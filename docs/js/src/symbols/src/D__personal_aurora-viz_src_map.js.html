1 AR.Canvas = function(graphDef, data, div) {
  2     this.data = data;
  3     this.graphDef = graphDef;
  4     this.div = div;
  5 };
  6 AR.Canvas.prototype = pv.extend(GOverlay);
  7 
  8 AR.Canvas.prototype.initialize = function(map) {
  9     this.map = map;
 10     this.canvas = document.createElement("div");
 11     this.canvas.setAttribute("class", "canvas");
 12     map.getPane(G_MAP_MAP_PANE).parentNode.appendChild(this.canvas);
 13 };
 14 AR.Canvas.prototype.setMarkerSize = function(size) {
 15     this._dot.shapeRadius(function(x, d) {
 16         var s = AR.constants.scale.mapDot * size * (d.size || 1);
 17         return s;
 18     });
 19 };
 20 
 21 AR.Canvas.prototype.setMarkerShape = function(shape) {
 22     var self = this;
 23     self._dot.shape(shape || "circle");
 24 };
 25 
 26 AR.Canvas.prototype.getCategoryColorMap = function() {
 27     var colors = AR.Utility.getPaletteColors(this.graphDef);
 28     var cLength = colors.length;
 29     var categoryColorMap = {};
 30     var i = 0,
 31         j = 0;
 32     for (i = 0; i < this.data.length; i++) {
 33         var c = this.data[i].category;
 34         if (!categoryColorMap[c]) {
 35             categoryColorMap[c] = colors[j % cLength];
 36             j++;
 37         }
 38     }
 39     return categoryColorMap;
 40 };
 41 AR.Canvas.prototype.setMarkerColors = function(colorMap) {
 42     this._dot.strokeStyle(function(x, d) {
 43         return colorMap[d.category];
 44     });
 45     this._dot.fillStyle(function(x, d) {
 46         return this.strokeStyle().alpha(0.2);
 47     });
 48 };
 49 AR.Canvas.prototype.redraw = function(force) {
 50     var self = this;
 51     if (!force) {
 52         return;
 53     }
 54     var c = this.canvas,
 55         m = this.map,
 56         r = 500;
 57 
 58     /* Get the pixel locations of the crimes. */
 59     var pixels = this.data.map(function(d) {
 60         return m.fromLatLngToDivPixel(new GLatLng(d.lat, d.lon));
 61     }); /* Update the canvas bounds. Note: may be large. */
 62 
 63     function getX(p) {
 64         return p.x;
 65     }
 66 
 67     function getY(p) {
 68         return p.y;
 69     }
 70 
 71     var x = {
 72         min: pv.min(pixels, getX) - r,
 73         max: pv.max(pixels, getX) + r
 74     };
 75     var y = {
 76         min: pv.min(pixels, getY) - r,
 77         max: pv.max(pixels, getY) + r
 78     };
 79 
 80     c.style.width = (x.max - x.min) + "px";
 81     c.style.height = (y.max - y.min) + "px";
 82     c.style.left = x.min + "px";
 83     c.style.top = y.min + "px";
 84 
 85     /* Render the visualization. */
 86     this._panel = new pv.Panel().canvas(c).left(-x.min).top(-y.min).add(pv.Panel).data(self.data);
 87 
 88     // Create the marker
 89     this._dot = this._panel.add(pv.Dot).left(function() {
 90         return pixels[this.parent.index].x;
 91     }).top(function() {
 92         return pixels[this.parent.index].y;
 93     });
 94 
 95     //set the title attribute
 96     self._dot.title(function(x, d) {
 97         return (d.lat + ", " + d.lon + ", " + d.category + ", ");
 98     });
 99 
100     this.setMarkerShape(this.graphDef.mapMarkerShape);
101     this.setMarkerSize(this.graphDef.mapMarkerSize);
102     this.setMarkerColors(this.getCategoryColorMap());
103     AR.Utility.setToolTip(this.graphDef, self._dot, "s");
104     if (this.graphDef.showLabels) {
105         var labels = this._dot.anchor("center").add(pv.Label).textStyle("white").text(function(x, d) {
106             return d.category;
107         });
108         AR.Utility.setLabelProperties(this.graphDef, labels, false);
109     }
110     this.render(this.div);
111 };
112 AR.Canvas.prototype.render = function(div) {
113     var mapDiv = document.createElement('div');
114     document.getElementById(div).appendChild(mapDiv);
115     this._panel.canvas(mapDiv);
116     this._panel.root.render();
117 };

121     var self = this;
122     self.render = function(div) {
123         if (GBrowserIsCompatible()) {

134             map.addOverlay(can);
