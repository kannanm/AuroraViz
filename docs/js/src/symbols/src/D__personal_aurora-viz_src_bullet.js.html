  4 /**
  5  * A constructor function to create a Bullet graph.
  6  *
  7  * @class
  8  *
  9  * @param {object}
 10  *            [graphDef] An object containing the graph properties and the data
 11  * @extends AR.Graph
 12  */
 13 
 14 AR.BulletGraph = function(graphDef) {
 15     var self = this;
 16     self._bullet = undefined;
 17     self._bulletPanel = undefined;
 18     self._graphDef = graphDef;
 19     //TODO: Create the following using function generator as there is a minor difference between the two
 20     self.initialize(graphDef);
 21 
 22     /**
 23      * @private
 24      * Creates a bullet panel and adds to the parent panel
 25      */
 26     var addBulletPanel = function() {
 27         self._bulletPanel = self._panel.add(pv.Panel).margin(20);
 28 
 29     };
 30 
 31     /**
 32      *  @private
 33      * Sets the top property of the bullet panel as it should not be placed one on the other
 34      */
 35     var setBulletPanelTop = function() {
 36         self._bulletPanel.top(function() {
 37             return this.index * 2 * (self._dimension.height) / (2 * self._graphDef.data.length) + 40;
 38         });
 39     };
 40 
 41     /**
 42      * @private
 43      * Creates the bullet layout and adds to the bulletpanel
 44      */
 45     var addBullets = function() {
 46         self._bullet = self._bulletPanel.add(pv.Layout.Bullet);
 47     };
 48 
 49     //  The following is an implementation of function generator that creates set<Property> functions.
 50     //    
 51     //    var props = ["ranges","measures","markers"];
 52     //    var setterFunc = function(){
 53     //        var retObject = {},i,property;
 54     //        for(i =0 ;i<props.length;i++){
 55     //            property = props[i];
 56     //            upcasedProp = property.substring(0, 1).toUpperCase() + property.substring(1);
 57     //            retObject["set"+upcasedProp] = function(){
 58     //                bullet[property](function(d) d[property]);
 59     //            }
 60     //        }
 61     //        retun retObject;
 62     //    };
 63     /**
 64      * @private
 65      * Sets the data for ranges field in bullet object
 66      */
 67     var setRange = function() {
 68         self._bullet.ranges(function(d) {
 69             return d.ranges;
 70         });
 71     };
 72 
 73     /**
 74      * @private
 75      * Sets the data for measures field in bullet object
 76      */
 77     var setMeasures = function() {
 78         self._bullet.measures(function(d) {
 79             return d.measures;
 80         });
 81     };
 82 
 83     /**
 84      * @private
 85      * Sets the data for markers field in bullet object
 86      */
 87     var setMarkers = function() {
 88         self._bullet.markers(function(d) {
 89             return d.markers;
 90         });
 91     };
 92 
 93     /**
 94      * @private
 95      *  Adds the range element
 96      */
 97     var addRangeElement = function(elem) {
 98         self._bullet.range.add(pv[elem] || pv.Bar);
 99     };
100 
101     /**
102      * @private
103      *  Adds the measures element
104      */
105     var addMeasureElement = function(elem) {
106         self._bullet.measure.add(pv[elem] || pv.Bar);
107     };
108 
109     /**
110      * @private
111      *  Adds the marker element
112      */
113     var addMarkerElement = function(elem) {
114         self._bullet.marker.add(pv[elem] || pv.Dot);
115     };
116 
117     /**
118      * @private
119      * Creates the rule for the bullet object according the the range  
120      */
121     var createRule = function() {
122         self._bullet.tick.add(pv.Rule).anchor("bottom").add(pv.Label).text(self._bullet.x.tickFormat);
123     };
124 
125     /**
126      * @private
127      * Adds text label to each bullet object as given in <b>title</b> property of the data
128      */
129     var addLabel = function() {
130         if (self._graphDef.showLabels) {
131             var labels = self._bullet.anchor("left").add(pv.Label).textBaseline("bottom").text(function(d) {
132                 return d.title;
133             });
134             AR.Utility.setLabelProperties(self._graphDef, labels, true);
135             labels.textAlign("right");
136         }
137     };
138 
139     /**
140      * @private
141      *  Sets the value
142      */
143     var addValues = function() {
144         if (self._graphDef.showValues) {
145             //TODO write code for functionality
146         }
147     };
148 
149     /**
150      * @private
151      *  Sets the title text as the toolTip text given in the data
152      */
153     var setTitleText = function() {
154         self._bullet.title(function(d) {
155             return d.toolTipText;
156         });
157     };
158 
159     /**
160      * @private
161      * A function to create the bullet chart. Its an aggregator function callin all the other functions
162      */
163     var createBulletChart = function() {
164         addBulletPanel();
165         self.addData(self._graphDef.data);
166         self.setBulletPanelWidth();
167         self.setBulletPanelHeight();
168         self.setBulletPanelLeft();
169         setBulletPanelTop();
170         addBullets();
171         self.setOrientation(self._graphDef.bulletOrientation);
172         setRange();
173         setMeasures();
174         setMarkers();
175         addRangeElement();
176         addMeasureElement();
177         addMarkerElement();
178         self.setMarkerShape(self._graphDef.markerShape);
179         self.setMarkerFillStyle(self._graphDef.markerFillColor);
180         self.setMeasuresFillStyle(self._graphDef.measureFillColor);
181         self.setRangeFillStyle();
182         createRule();
183         addLabel();
184         setTitleText();
185         AR.Utility.setToolTip(self._graphDef, self._bullet, "s");
186     };
187     createBulletChart();
188 };
189 
190 AR.BulletGraph.prototype = new AR.Graph();;
191 
192 AR.BulletGraph.prototype.setVerGridShow = function(status) {
193     var self = this;
194     if (status === true) {
195         AR.Graph.prototype.setVerGridShow.apply(self, [self._dimension.width, AR.Utility.scale.linear]);
196     }
197 };
198 AR.BulletGraph.prototype.setHorGridShow = function(status) {
199     var self = this;
200     if (status === true) {
201         AR.Graph.prototype.setHorGridShow.apply(self, [self._dimension.height, AR.Utility.scale.linear]);
202     }
203 };

205 /**
206  * adds Data to the bullet panel
207  * @param {object}
208  *           [data] Data to be added to bullet Panel
209  */
210 AR.BulletGraph.prototype.addData = function(data) {
211     var self = this;
212     self._bulletPanel.data(data);
213 };
214 
215 /**
216  * sets the width of the bullet panel
217  * @param {number}
218  *           [width] The width of the bullet panel
219  */
220 AR.BulletGraph.prototype.setBulletPanelWidth = function(width) {
221     var self = this;
222     self._bulletPanel.width(width || (self._dimension.width - 40) / 2);
223 };
224 
225 /**
226  * sets the height of the bullet panel
227  * @param {number}
228  *           [height] The height of the bullet panel
229  */
230 AR.BulletGraph.prototype.setBulletPanelHeight = function(height) {
231     var self = this;
232     self._bulletPanel.height(height || (self._dimension.height - 40) / (2 * self._graphDef.data.length));
233 };
234 
235 /**
236  * sets the left property of the bullet panel object
237  * @param {number}
238  *           [left] The left of the bullet panel
239  */
240 AR.BulletGraph.prototype.setBulletPanelLeft = function(left) {
241     var self = this;
242     self._bulletPanel.left(left || 100);
243 };
244 
245 /**
246  * sets the orientation of the bullet. It can be set to either left or right
247  * @param {string}
248  *           [orientation] The orientation of the bullet
249  */
250 AR.BulletGraph.prototype.setOrientation = function(orientation) {
251     var self = this;
252     self._bullet.orient(orientation || "left");
253 };
254 
255 /**
256  * sets the shape of the marker. It can be set to circle, square, triangle etc.
257  * @param {string}
258  *           [shape] The marker shape  
259  */
260 AR.BulletGraph.prototype.setMarkerShape = function(shape) {
261     var self = this;
262     self._bullet.marker.shape(shape || "triangle");
263 };
264 
265 /**
266  * sets the fill color of the marker.
267  * @param {string}
268  *           [color] The marker fillcolor  
269  */
270 AR.BulletGraph.prototype.setMarkerFillStyle = function(color) {
271     var self = this;
272     self._bullet.marker.fillStyle(color || "white");
273 };
274 
275 /**
276  * sets the fill color of the marker.
277  * @param {string}
278  *           [color] The marker fillcolor  
279  */
280 AR.BulletGraph.prototype.setMeasuresFillStyle = function(color) {
281     var self = this;
282     self._bullet.measure.fillStyle(color || "#1F77B4");
283 };
284 
285 
286 /**
287  * sets the fill color of the range accordign to the palette selected.
288  */
289 
290 AR.BulletGraph.prototype.setRangeFillStyle = function() {
291     var self = this;
292     var colors = AR.Utility.getPaletteColors(self._graphDef);
293     self._bullet.range.fillStyle(function() {
294         return colors[this.index % colors.length];
295     });
296 };
