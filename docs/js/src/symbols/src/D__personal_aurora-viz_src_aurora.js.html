 10 /**
 11  
 12  * Returns a prototype object suitable for extending the
 13  *         given class <tt>f</tt>. For more details, see Douglas Crockford's
 14  *         essay on prototypal inheritance.
 15  * @param  {function}
 16  *            f a constructor.
 17  * @returns a suitable prototype object.
 18  * @see Douglas Crockford's essay on <a
 19  *      href="http://javascript.crockford.com/prototypal.html">prototypal
 20  *      inheritance</a>.
 21  */
 22 AR.extend = function (f) {
 23   function g() {}
 24   g.prototype = f.prototype || f;
 25   return new g();
 26 };

 28 /**
 29  * @class
 30  * The base class for the graph initilization.
 31  * Initializes the panel and sets parentDimensions and property values
 32  */
 33 AR.Graph = function () {
 34   var self = this;
 35   self._dimension = {
 36     height: 400,
 37     width: 400,
 38     left: 60,
 39     right: 20,
 40     bottom: 60,
 41     top: 60
 42   };
 43   self._graphDef = undefined;
 44   self._pos = undefined;
 45   self._caption = undefined;
 46   self._horAxisLabel = undefined;
 47   self._verAxisLabel = undefined;
 48   self._horGrid = undefined;
 49   self._verGrid = undefined;
 50   self._outerPanel = undefined;
 51   self._panel = undefined;
 52   self._horAxis = undefined;
 53   self._verAxis = undefined;
 54 };

 56 /**
 57  * The function initializes the panel and sets the properties which are common to all graphs
 58  * @param {Object} graphDef The graph definition containing styling, data and chart speficig properties  
 59  */
 60 AR.Graph.prototype.initialize = function (graphDef) {
 61   var self = this;
 62   self._outerPanel = new pv.Panel().left(0).right(0).bottom(0).top(0);
 63   self._panel = self._outerPanel.add(pv.Panel);
 64   self._panel.left(self._dimension.left).right(self._dimension.right).bottom(self._dimension.bottom).top(self._dimension.top);
 65   self._horAxis = self._panel.add(pv.Rule).bottom(0);
 66   self._verAxis = self._panel.add(pv.Rule).left(0);
 67   self._graphDef = graphDef;
 68   var properties = ["width", "height", "caption", "xAxisName", "yAxisName"];
 69   var upcasedProp;
 70   properties.forEach(function (property) {
 71     upcasedProp = property.substring(0, 1).toUpperCase() + property.substring(1);
 72     if (graphDef[property]) {
 73       self["set" + upcasedProp](graphDef[property]);
 74     }
 75   });
 76   var styles = graphDef.style;
 77   var prop;
 78   for (prop in styles) {
 79     upcasedProp = prop.substring(0, 1).toUpperCase() + prop.substring(1);
 80     if (styles.hasOwnProperty(prop)) {
 81       self["set" + upcasedProp](styles[prop]);
 82     }
 83   }
 84 };

 86 /**
 87  * Sets a graph's caption.
 88  * Caption is one of the properties of any graph
 89  * @param {string}
 90  *             Sets the caption for the given graph as the supplied parameter value
 91  */
 92 AR.Graph.prototype.setCaption = function (newVal) {
 93   var self = this;
 94   if (self._caption) {
 95     self._caption.text(newVal);
 96   } else {
 97     self._caption = self._outerPanel.add(pv.Label).text(newVal).left(self._dimension.width / 2 + 40).top(0).textAlign("center").textBaseline("top");
 98   }
 99 };
100 
101 /**
102  * Sets a graph's width.
103  * Width is one of the properties of a graph
104  * @param width Width for the given graph as the supplied parameter value
105  *            
106  */
107 AR.Graph.prototype.setWidth = function (width) {
108   var self = this;
109   self._dimension.width = width - self._dimension.left - self._dimension.right;
110   self._outerPanel.width(width);
111   if (self._caption) {
112     self._caption.left(self._dimension.width / 2);
113   }
114 };

116 /**
117  * Sets a graph's Height.
118  * @param height Height of the graph
119  */
120 AR.Graph.prototype.setHeight = function (height) {
121   var self = this;
122   self._dimension.height = height - self._dimension.top - self._dimension.bottom;
123   self._outerPanel.height(height);
124 };

154 /**
155  * Sets horizontal rules in a graph.
156  * Rules are nothing but lines that divide the graph to form a grid like structure
157  * @param maxValue {integer}
158  *                This is the maximimum value that the graph would have in horizontal direction. It is required to calculate the scale.
159  * @param scaleType {object}
160  *             It takes the scaleType (Linear/Ordinal etc)
161  */
162 
163 AR.Graph.prototype.setHorGridShow = function (maxValue, scaleType) {
164   var self = this;
165   self._y = pv.Scale[scaleType](0, maxValue).range(0, self._dimension.height - 40);
166   if (!self._horGrid) {
167     self._horGrid = self._panel.add(pv.Rule);
168     self._horGrid.data(self._y.ticks().splice(1, self._y.ticks().length)).bottom(self._y);
169   } else {
170     self._horGrid.bottom(self._y);
171   }
172 };
173 
174 /**
175  * It shows the Horizontal grid axis if the <b>status</b> is set to true
176  * @param status {boolean}
177  *          If set to true it show the labels for the Horizontal grid
178  */
179 AR.Graph.prototype.setHorGridLabelShow = function (status) {
180   var self = this;
181   if (self._horGrid && status === true) {
182     self._horGrid.anchor("left").add(pv.Label).text(self._y.tickFormat).font(AR.Utility.getSize(self._graphDef, "gridLabels", 15) + "px Arial");
183   }
184 };

219 /**
220  * Renders the Component
221  * @param div {html node}
222  *             The node in which the graph has to be rendered. If undefined grpah is rendered whereever the function is called.
223  */
224 AR.Graph.prototype.render = function render(div) {
225   var self = this;
226   if (div || self.__pos) {
227     self._outerPanel.canvas(div || self.__pos);
228     self._outerPanel.render();
229     self.__pos = div || self.__pos;
230   } else {
231     self._outerPanel.render();
232   }
233 };

293 /**
294  * The function creates functions which set the <tt>property</tt> of the elements in the <tt>elementsArr</tt>
295  * @param elementsArr {Array} The array of elements whos <tt>property</tt> is to be set
296  * @param property {String} The property which is to be set 
297  * @param protovisProp {String} The corresponding protovis property
298  * @param additionalData {String} Some additional data to set the property. 
299  */
300 AR.Graph.createFunctions = function (elementsArr, property, protovisProp, additionalData) {
301   var i = 0;
302   for (i = 0; i < elementsArr.length; i++) {
303     var prop = elementsArr[i].name;
304     var upcasedProp = prop.substring(0, 1).toUpperCase() + prop.substring(1);
305     var obj = elementsArr[i].obj;
306     AR.Graph.prototype["set" + upcasedProp + property] = (function (prop, obj) {
307       return function (value) {
308         var o = this[obj];
309         if (o) {
310           o[protovisProp](value + (additionalData || ""));
311         }
312       };
313     })(prop, obj);
314   }
315 };
