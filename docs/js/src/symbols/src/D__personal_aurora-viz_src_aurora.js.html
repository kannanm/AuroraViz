  2 Array.prototype.max = Array.prototype.max ||
  3 function () {
  4   var max = this[0];
  5   var len = this.length;
  6   var i;
  7   for (i = 1; i < len; i++) {
  8     if (this[i] > max) {
  9       max = this[i];
 10     }
 11   }
 12   return max;
 13 };

 41 /**
 42  * @class
 43  * The basefunction for the graph initilization.
 44  * Initializes the panel and sets parentDimensions and property values
 45  * @param {json}
 46  *             graphDef is the current graph Dimension
 47  * This method should be the first method that should be called when creating any kind of graph.
 48  * Creates the panel, sets its dimensions, labels and property values and returns.
 49  */
 50 AR.Graph = function () {
 51   var self = this;
 52   self._dimension = {
 53     height: 400,
 54     width: 400,
 55     left: 60,
 56     right: 20,
 57     bottom: 60,
 58     top: 60
 59   };
 60   self._graphDef = undefined;
 61   self._pos = undefined;
 62   self._caption = undefined;
 63   self._horAxisLabel = undefined;
 64   self._verAxisLabel = undefined;
 65   self._horGrid = undefined;
 66   self._verGrid = undefined;
 67   self._outerPanel = undefined;
 68   self._panel = undefined;
 69   self._horAxis = undefined;
 70   self._verAxis = undefined;
 71 };
 72 
 73 AR.Graph.prototype.initialize = function (graphDef) {
 74   var self = this;
 75   self._outerPanel = new pv.Panel().left(0).right(0).bottom(0).top(0);
 76   self._panel = self._outerPanel.add(pv.Panel);
 77   self._panel.left(self._dimension.left).right(self._dimension.right).bottom(self._dimension.bottom).top(self._dimension.top);
 78   self._horAxis = self._panel.add(pv.Rule).bottom(0);
 79   self._verAxis = self._panel.add(pv.Rule).left(0);
 80   self._graphDef = graphDef;
 81   var properties = ["width", "height", "caption", "xAxisName", "yAxisName"];
 82   var upcasedProp;
 83   properties.forEach(function (property) {
 84     upcasedProp = property.substring(0, 1).toUpperCase() + property.substring(1);
 85     if (graphDef[property]) {
 86       self["set" + upcasedProp](graphDef[property]);
 87     }
 88   });
 89   var styles = graphDef.style;
 90   var prop;
 91   for (prop in styles) {
 92     upcasedProp = prop.substring(0, 1).toUpperCase() + prop.substring(1);
 93     if (styles.hasOwnProperty(prop)) {
 94       self["set" + upcasedProp](styles[prop]);
 95     }
 96   }
 97 };

 99 /**
100  * Sets a graph's caption.
101  * Caption is one of the properties of any graph
102  * @param {string}
103  *             Sets the caption for the given graph as the supplied parameter value
104  */
105 AR.Graph.prototype.setCaption = function (newVal) {
106   var self = this;
107   if (self._caption) {
108     self._caption.text(newVal);
109   } else {
110     self._caption = self._outerPanel.add(pv.Label).text(newVal).left(self._dimension.width / 2 + 40).top(0).textAlign("center").textBaseline("top");
111   }
112 };
113 
114 /**
115  * Sets a graph's width.
116  * Width is one of the properties of a graph
117  * @param {string}
118  *             [width] Sets the width for the given graph as the supplied parameter value
119  */
120 AR.Graph.prototype.setWidth = function (width) {
121   var self = this;
122   self._dimension.width = width - self._dimension.left - self._dimension.right;
123   self._outerPanel.width(width);
124   if (self._caption) {
125     self._caption.left(self._dimension.width / 2);
126   }
127 };

135   var self = this;

168 /**
169  * Sets horizontal rules in a graph.
170  * Rules are nothing but lines that divide the graph to form a grid like structure
171  * @param {integer}
172  *               [maxValue] This is the maximimum value that the graph would have in horizontal direction. It is required to calculate the scale.
173  * @param {object}
174  *            [scaleType] It takes the scaleType (Linear/Ordinal etc)
175  */
176 
177 AR.Graph.prototype.setHorGridShow = function (maxValue, scaleType) {
178   var self = this;
179   self._y = pv.Scale[scaleType](0, maxValue).range(0, self._dimension.height - 40);
180   if (!self._horGrid) {
181     self._horGrid = self._panel.add(pv.Rule);
182     self._horGrid.data(self._y.ticks().splice(1, self._y.ticks().length)).bottom(self._y);
183   } else {
184     self._horGrid.bottom(self._y);
185   }
186 };

294 AR.Graph.createFunctions = function (elementsArr, property, protovisProp, additionalData) {
295   var i = 0;
296   for (i = 0; i < elementsArr.length; i++) {
297     var prop = elementsArr[i].name;
298     var upcasedProp = prop.substring(0, 1).toUpperCase() + prop.substring(1);
299     var obj = elementsArr[i].obj;
300     AR.Graph.prototype["set" + upcasedProp + property] = (function (prop, obj) {
301       return function (value) {
302         var o = this[obj];
303         if (o) {
304           o[protovisProp](value + (additionalData || ""));
305         }
306       };
307     })(prop, obj);
308   }
309 };
