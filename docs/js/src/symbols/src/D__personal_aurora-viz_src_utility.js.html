  3 /**
  4  * Finds the maximum value in the array
  5  */
  6 Array.prototype.max = Array.prototype.max ||
  7 function() {
  8     var max = this[0];
  9     var len = this.length;
 10     var i;
 11     for (i = 1; i < len; i++) {
 12         if (this[i] > max) {
 13             max = this[i];
 14         }
 15     }
 16     return max;
 17 };

 35 /**
 36  * Function returns the keys for an object. Check if the the function already exist if not then create the function.
 37  * The implementation has been taken from <a href="http://stackoverflow.com/questions/208016/how-to-list-the-properties-of-a-javascript-object#answer-3937321">here</a>.
 38  */
 39 AR.Utility.getKeys = Object.keys ||
 40 function() {
 41     var ownProperty = Object.prototype.hasOwnProperty,
 42         hasDontEnumBug = !{
 43             toString: null
 44         }.propertyIsEnumerable("toString"),
 45         DontEnums = [
 46             'toString', 'toLocaleString', 'valueOf', 'hasOwnProperty',
 47             'isPrototypeOf', 'propertyIsEnumerable', 'constructor'
 48             ],
 49         DontEnumsLength = DontEnums.length;
 50     return function(o) {
 51         if (typeof(o !== "object" && typeof o !== "function") || o === null) {
 52             throw new TypeError("Object.keys called on a non-object");
 53         }
 54         var result = [],
 55             name, i;
 56         for (name in o) {
 57             if (ownProperty.call(o, name)) {
 58                 result.push(name);
 59             }
 60         }
 61 
 62         if (hasDontEnumBug) {
 63             for (i = 0; i < DontEnumsLength; i++) {
 64                 if (ownProperty.call(o, DontEnums[i])) {
 65                     result.push(DontEnums[i]);
 66                 }
 67             }
 68         }
 69         return result;
 70     };
 71 };

 91 /**
 92  * Finds the maximum value in data Object
 93  * @param data It is an array of Objects  
 94  * @returns the maximum value
 95  */
 96 AR.Utility.findMax = function(data) {
 97     var max = data[0].value;
 98     data.forEach(function(data) {
 99         if (parseInt(data.value, 10) > max) {
100             max = parseInt(data.value, 10);
101         }
102     });
103     return max;
104 };

108     var max = parseInt(data[0], 10);
109     var i = 0;
110     for (i = 0; i < data.length; i++) {
111         var cur = parseInt(data[i], 10);
112         if (cur > max) {
113             max = cur;
114         }
115     }
116     return max;

260 /**
261  * Finds the maximum value from the measure axis
262  * @param graphDef the graph defination
263  * @returns the maximum value
264  */
265 AR.Utility.findMaxValue = function(graphDef) {
266     if (graphDef.dataset) {
267         var dataArray = [];
268         for (i = 0; i < graphDef.dataset.length; i++) {
269             var max = AR.Utility.findMax(graphDef.dataset[i].data);
270             var maxMap = {
271                 "value": max
272             };
273             dataArray.push(maxMap);
274         }
275         return AR.Utility.findMax(dataArray);
276     } else {
277         return AR.Utility.findMax(graphDef.data);
278     }
279 };

319 /**
320  *  Sets the rotation angle of the <tt>labelObj</tt>.
321  * @param angle The angle in degrees
322  * @param labelObj The label object whose rotation angle is to be set
323  * @param graphDef The graph Definition
324  * @param isHorizontal true if the graph is horizontal, false otherwise
325  */
326 AR.Utility.setLabelRotateAngle = function(angle, labelObj, graphDef, isHorizontal) {
327     var dummy;
328     angle = parseInt(angle, 10);
329     if (labelObj) {
330         labelObj.textAngle(angle / 180 * Math.PI);
331         if (isHorizontal) {
332             dummy = angle < 0 ? labelObj.textBaseline("right") : angle === 0 ? labelObj.textBaseline("center") : labelObj.textBaseline("top");
333             dummy = (angle === 90 || angle === -90) ? labelObj.textAlign("center") : labelObj.textAlign("right");
334         } else {
335             dummy = angle < 0 ? labelObj.textAlign("right") : angle === 0 ? labelObj.textAlign("center") : labelObj.textAlign("left");
336         }
337 
338     }
339 };

357 /**
358  * Create legend and add it to the panel
359  * @param panel the panel to which the legends will be added
360  * @param index the index of the legend
361  * @param text the text for the legend
362  * @param color the color of the legend
363  * @param graphDef the graph definition
364  */
365 AR.Utility.createLegends = function(panel, index, text, color, graphDef) {
366     if (text) {
367         var dots = panel.add(pv.Dot).shape("square").top(AR.constants.values.legend.top).fillStyle(color).shapeRadius(AR.Utility.getSize(graphDef, "legend", AR.constants.values.legend.size));
368         AR.Utility.setLeftOfLegends(dots, graphDef, index);
369         dots.anchor("right").add(pv.Label).text(function() {
370             return text;
371         }).font(AR.Utility.getSize(graphDef, "gridLabels", AR.constants.values.smallLabels.size) + "px Arial");
372     }
373 };

439 /**
440  * Calculates the size of the <tt>element</tt> with respect to the height and width of the graph and the actual <tt>value</tt>
441  * @param graphDef the graph definition
442  * @param element the element whose height is required
443  * @param value The value set by the user
444  * @returns {Number} The size of the <tt>element</tt>
445  */
446 AR.Utility.getSize = function(graphDef, element, value) {
447     return AR.constants.scale[element] * value * graphDef.height * graphDef.width;
