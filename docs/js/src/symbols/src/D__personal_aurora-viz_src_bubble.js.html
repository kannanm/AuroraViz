  3 /**
  4  * The basefunction for the Bubble initialization. Initializes the BubbleGraph.
  5  * Sets the data
  6  * 
  7  * @params {Object} [graphDef] graphDef object supplied by the User for which
  8  *         the graph needs to be plotted. It contains graph properties and the
  9  *         data
 10  * @params {Object} [parentDimension] parentDimension object that will indicate
 11  *         the dimensions of AR.Graph
 12  * @params {Object} [panel] A panel object indicating the Graph Panel in which
 13  *         the current Bubble graph will be displayed This is a base class and
 14  *         is called by different Bubble implementations
 15  */
 16 AR.Bubble = function(parentDimension, panel, graphDef) {
 17 	var self = this;
 18 	var bubble = panel.add(pv.Dot);
 19 	bubble.data(AR.Utility.getMultiDimensionData(graphDef.data));
 20 	/**
 21 	 * @private set the bottom property of the bubble object. The bottom
 22 	 *          property depends on the "y" key in the data
 23 	 */
 24 	var adjustBottom = function(parentDimension) {
 25 		var maxVal = AR.Utility.getSingleDimensionData(graphDef.data, AR.Utility.Dimension.y).max();
 26 		var bubbleBottom = pv.Scale.linear(0, maxVal).range(0, parentDimension.height - 40);
 27 		bubble.bottom(function(d) {
 28 			return bubbleBottom(d[1]);
 29 		});
 30 	};
 31 
 32 	/**
 33 	 * @private set the left property of the bubble object. The left property
 34 	 *          depends on the "x" key in the data
 35 	 */
 36 	var adjustLeft = function(parentDimension) {
 37 		var maxVal = AR.Utility.getSingleDimensionData(graphDef.data, AR.Utility.Dimension.x).max();
 38 		var bubbleLeft = pv.Scale.linear(0, maxVal).range(0, parentDimension.width - 30);
 39 		bubble.left(function(d) {
 40 			return (bubbleLeft(d[0]));
 41 		});
 42 	};
 43 
 44 	/**
 45 	 * @private set the size property of the bubble object. The size property
 46 	 *          depends on the "z" key in the data
 47 	 */
 48 	var adjustSize = function(parentDimension) {
 49 		var maxVal = AR.Utility.getSingleDimensionData(graphDef.data, AR.Utility.Dimension.z).max();
 50 		var smallerDim = Math.min(parentDimension.width, parentDimension.height);
 51 		var bubbleRadius = pv.Scale.linear(0, maxVal).range(0, smallerDim / 20);
 52 		bubble.shapeRadius(function(d) {
 53 			return (bubbleRadius(d[2]));
 54 		});
 55 	};
 56 	bubble.title(function(d) {
 57 		return AR.Utility.getToolTipText(graphDef.data, this.index) || (graphDef.data[this.index].label + ", " + d[0] + ", " + d[1] + ", " + d[2]);
 58 	});
 59 	/**
 60 	 * @private Sets the tooltip if graphDef.toolTip is true
 61 	 */
 62 	var showToolTip = function() {
 63 		if (graphDef.toolTip) {
 64 			AR.Utility.setToolTip(graphDef, bubble, "s");
 65 		}
 66 	};
 67 
 68 	/**
 69 	 * @private Creates the label for bubble object if graphDef.showLabels is
 70 	 *          true
 71 	 */
 72 	var showLabel = function() {
 73 		if (graphDef.showLabels) {
 74 			var labels = bubble.anchor("bottom").add(pv.Label).text(function(d) {
 75 				return graphDef.data[this.index].label;
 76 			});
 77 			AR.Utility.setLabelProperties(graphDef, labels, false);
 78 
 79 		}
 80 	};
 81 
 82 	/**
 83 	 * @private 
 84 	 * Shows the value for the bubble object if graphDef.showValues is
 85 	 * true
 86 	 */
 87 	var showValues = function() {
 88 		if (graphDef.showValues) {
 89 			var values = bubble.anchor("top").add(pv.Label).text(function(d) {
 90 				return (d[2]);
 91 			});
 92 			AR.Utility.setLabelProperties(graphDef, values, false);
 93 		}
 94 	};
 95 	
 96 	/**
 97 	 * @private
 98 	 * Sets the palette for the bubble object
 99 	 */
100 	var setPalette = function() {
101 		if (graphDef.presetPalette) {
102 			AR.Utility.setPalette(bubble, AR.Utility.getPaletteColors(graphDef));
103 		}
104 	};
105 	
106 	/**
107 	 * @private
108 	 * Creates the legend for the Bubble
109 	 */
110 	var showLegends = function() {
111 		if (graphDef.showLegends) {
112 			var i = 0;
113 			var data = graphDef.data;
114 			var colors = AR.Utility.getPaletteColors(graphDef);
115 			for (i = 0; i < data.length; i++) {
116 				AR.Utility.createLegends(panel, i, data[i].label, colors[i % data.length], graphDef);
117 			}
118 		}
119 	};
120 	/**
121 	 * Adjusts the position of the bubbles according to the
122 	 * <tt>parentDimension</tt> which the dimension of the chart
123 	 * 
124 	 * @param parentDimension
125 	 *            {Object} The dimension of the chart
126 	 */
127 	self.adjustPosition = function(parentDimension) {
128 		adjustBottom(parentDimension);
129 		adjustLeft(parentDimension);
130 		adjustSize(parentDimension);
131 		showToolTip();
132 		showLabel();
133 		showValues();
134 		setPalette();
135 		showLegends();
136 	};
137 	self.adjustPosition(parentDimension);
138 };

140 /**
141  * @class
142  * API to construct a Bubble Graph
143  * 
144  * @param {object} An object containing the graph properties and the data
145  * @extends AR.Graph
146  */
147 AR.BubbleGraph = function(graphDef) {
148 	var self = this;
149 	/**
150 	 * If <tt>status</tt> is true show the vertical grid
151 	 * 
152 	 * @param status
153 	 *            {Boolean} true if vertical grid has to be visible. False
154 	 *            otherwise
155 	 */
156 	self.setVerGridShow = function(status) {
157 		if (status === true) {
158 			AR.Graph.prototype.setVerGridShow.apply(self, [ AR.Utility.getSingleDimensionData(graphDef.data, AR.Utility.Dimension.y).max(), AR.Utility.scale.linear ]);
159 		}
160 	};
161 	/**
162 	 * If <tt>status</tt> is true show the horizontal grid
163 	 * 
164 	 * @param status
165 	 *            {Boolean} true if horizontal grid has to be visible. False
166 	 *            otherwise
167 	 */
168 	self.setHorGridShow = function(status) {
169 		if (status === true) {
170 			AR.Graph.prototype.setHorGridShow.apply(self, [ AR.Utility.getSingleDimensionData(graphDef.data, AR.Utility.Dimension.x).max(), AR.Utility.scale.linear ]);
171 		}
172 	};
173 	self.initialize(graphDef);
174 
175 	// The bubble object
176 	var bubbles = new AR.Bubble(self._dimension, self._panel, graphDef);
177 
178 	/**
179 	 * Set the width of the graph
180 	 * 
181 	 * @param width
182 	 *            The width of the graph
183 	 */
184 	self.setWidth = function(width) {
185 		AR.Graph.prototype.setWidth.call(self, width);
186 		bubbles.adjustPosition(self._dimension);
187 	};
188 	/**
189 	 * Set the height of the graph
190 	 * 
191 	 * @param height
192 	 *            The height of the graph
193 	 */
194 	self.setHeight = function(height) {
195 		AR.Graph.prototype.setHeight.call(self, height);
196 		bubbles.adjustPosition(self._dimension);
197 	};
198 };
199 AR.BubbleGraph.prototype = new AR.Graph();
