  2 /**
  3  * The basefunction for the Node Link initialization.
  4  * Initializes the Node Link. Sets the data
  5  * @params {Object}
  6  *             [graphDef] graphDef object supplied by the User for which the graph needs to be plotted. It contains graph properties and the data
  7  * @params {Object}
  8  *             [parentDimension] parentDimension object that will indicate the dimensions of AR.Graph
  9  * @params {Object}
 10  *             [panel] A panel object indicating the Graph Panel in which the current Line graph will be displayed
 11  * This is a base class and is called by different Line implementations.
 12  */
 13 AR.NodeLinkGraphDefault = function(parentDimension, panel, graphDef) {
 14     var self = this;
 15     var nodeLinkPanel = panel;
 16     var layout = nodeLinkPanel.add(pv.Layout.Cluster);
 17     layout.nodes(pv.dom(graphDef.data).root(graphDef.root).sort(function(a, b) {
 18         return (pv.naturalOrder(a.nodeName, b.nodeName));
 19     }).nodes());
 20 
 21     self.setLayoutNodes = function(parentDimension) {
 22         layout.group(true);
 23         layout.orient("left");
 24     };
 25 
 26     var linkLayout = layout.link.add(pv.Line);
 27 
 28     self.setLayoutLink = function(parentDimension) {
 29         linkLayout.strokeStyle("#ccc");
 30         linkLayout.lineWidth(1);
 31         linkLayout.antialias(false);
 32     };
 33 
 34     var nodeLink = layout.node.add(pv.Dot);
 35 
 36     self.setDotFillStyle = function(graphDef) {
 37         nodeLink.fillStyle(function(n) {
 38             return (n.firstChild ? "#aec7e8" : "#ff7f0e");
 39         });
 40     };
 41 
 42     self.setLayoutLabel = function(parentDimension) {
 43         layout.label.add(pv.Label);
 44     };
 45 
 46     self.setDendogramProperties = function() {
 47         self.setLayoutNodes(parentDimension);
 48         self.setLayoutLink(parentDimension);
 49         self.setDotFillStyle(graphDef);
 50         self.setLayoutLabel(parentDimension);
 51     };
 52 
 53     self.setDendogramProperties();
 54 
 55 };

141     };
142 
143     self.setSunBurstSize = function() {
144         self.sunBurstPartition.size(function(d) {
145             return 12;
146         });
147     };
148 
149     self.setSunBurstOrder = function(flag) {
150         if (flag === true) {
151             self.sunBurstPartition.order("ascending");
152         } else {
153             self.sunBurstPartition.order("descending");
154         }
155     };
156 
157     self.setOrientation = function(orientationType) {
158         if (orientationType === null) {
159             // Setting default orientation..
160             self.sunBurstPartition.orient("radial");
161         } else {
162             self.sunBurstPartition.orient(orientationType);
163         }
164     };
165 
166 
167 
168     self.setFillStyle = function() {
169         partitionNodes.fillStyle(pv.Colors.category19().by(function(d) {
170             return (d.parentNode && d.parentNode.nodeName);
171         }));
172     };
173

175         if (style === null) {
176             partitionNodes.strokeStyle("#fff");
177         } else {
