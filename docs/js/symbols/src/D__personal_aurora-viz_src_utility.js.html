 15 AR.Utility.findMax = function(data) {
 16 	var max = data[0].value;
 17 	data.forEach(function(data) {
 18 		if (parseInt(data.value, 10) > max) {
 19 			max = parseInt(data.value, 10);
 20 		}
 21 	});
 22 	return max;
 23 };

 25 AR.Utility.findArrayMax = function(data) {
 26 	var max = parseInt(data[0], 10);
 27 	var i = 0;
 28 	for (i = 0 ; i < data.length ; i++){
 29 		var cur = parseInt(data[i], 10); 
 30 		if ( cur > max) {
 31 			max = cur;
 32 		}
 33 	}
 34 	return max;
 35 };

145 AR.Utility.findMaxValue = function(graphDef) {
146 	if (graphDef.dataset) {
147 		var dataArray = [];
148 		for (i = 0; i < graphDef.dataset.length; i++) {
149 			var max = AR.Utility.findMax(graphDef.dataset[i].data);
150 			var maxMap = {
151 				"value" : max
152 			};
153 			dataArray.push(maxMap);
154 		}
155 		return AR.Utility.findMax(dataArray);
156 	} else {
157 		return AR.Utility.findMax(graphDef.data);
158 	}
159 };

179 AR.Utility.setLabelRotateAngle = function(angle, labelObj, isHorizontal) {
180 	if (labelObj) {
181 		labelObj.textAngle(angle / 180 * Math.PI);
182 		if (isHorizontal) {
183 			angle < 0 ? labelObj.textBaseline("right") : angle == 0 ? labelObj.textBaseline("center") : labelObj.textBaseline("top");
184 			(angle == 90 || angle == -90) ? labelObj.textAlign("center") : labelObj.textAlign("right");
185 		} else {
186 			angle < 0 ? labelObj.textAlign("right") : angle == 0 ? labelObj.textAlign("center") : labelObj.textAlign("left");
187 		}
188 
189 	}
190 };

202 AR.Utility.getRGB= function(color){
203 	return {
204 		red:  parseInt(color.substring(1,3),16),
205 		green: parseInt(color.substring(3,5),16),
206 		blue : parseInt(color.substring(5,7),16),
207 	};
208 };
209 AR.Utility.addRGB = function(rgb,shift){
210 	rgb.red += shift;
211 	rgb.green += shift;
212 	rgb.blue += shift;
213 	return rgb;
214 };
215 AR.Utility.getHex = function(rgb){
216 	  var hexColor = rgb.blue | (rgb.green << 8) | (rgb.red << 16);
217 	  return "#" + hexColor.toString(16);
218 };
219 AR.Utility.getDarkerColor = function(color,shift){
220 	var rgb = AR.Utility.getRGB(color); 
221 	rgb = AR.Utility.addRGB(rgb,-shift);
222 	return AR.Utility.getHex(rgb); 
223 };
