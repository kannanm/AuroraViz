  3 /**
  4  * The basefunction for the Wedge initialization.
  5  * Initializes the WedgeGraph. Sets the data
  6  * @params {Object}
  7  *             [graphDef] graphDef object supplied by the User for which the graph needs to be plotted. It contains graph properties and the data
  8  * @params {Object}
  9  *             [parentDimension] parentDimension object that will indicate the dimensions of AR.Graph
 10  * @params {Object}
 11  *             [panel] A panel object indicating the Graph Panel in which the current Wedge graph will be displayed
 12  * This is a base class and is called by different Wedge implementations (Eg : Donut or Pie)
 13  */
 14 
 15 //TODO : Merge the piegraph and the donut graph because the only difference is having an inner radius
 16 AR.Wedge = function(parentDimension, panel, graphDef) {
 17     var properties = ["values", "labels", "legends"];
 18     var wedge = panel.add(pv.Wedge);
 19     var wedgeLabels, valueLabels;
 20     var dataValues = AR.Utility.getDataArray(graphDef.data);
 21     var self = this;
 22     var adjustRadius = function(parentDimension) {
 23         if (graphDef.pieRadius !== null) {
 24             wedge.outerRadius(AR.Utility.getSize(graphDef,"wedge",graphDef.pieRadius));
 25         } else {
 26             wedge.pieRadius(function() {
 27                 return (parentDimension.width < parentDimension.height ? (parentDimension.width - 30) / 2 : (parentDimension.height - 40) / 2);
 28             });
 29         }
 30     };
 31 
 32     var adjustAngle = function(parentDimension) {
 33         wedge.angle(function(d) {
 34             return ((d) * 2 * Math.PI);
 35         });
 36     };
 37 
 38     var adjustLabelPosition = function(parentDimension, labelFontSize, isValueLabel) {
 39         var labels, shift;
 40         if (isValueLabel) {
 41             labels = valueLabels;
 42             shift = labelFontSize;
 43         } else {
 44             labels = wedgeLabels;
 45             shift = 0;
 46         }
 47         if (labels) {
 48             labels.left(function() {
 49                 return (wedge.outerRadius() / 2 * Math.cos(wedge.midAngle()) + (parentDimension.width) / 2);
 50             }).bottom(function() {
 51                 return (-wedge.outerRadius() / 2 * Math.sin(wedge.midAngle()) + (parentDimension.height) / 2 - shift);
 52             });
 53         }
 54     };
 55 
 56     self.adjustPosition = function(parentDimension) {
 57         adjustRadius(parentDimension);
 58         adjustAngle(parentDimension);
 59         wedge.def("o", -1).left(function() {
 60             return (parentDimension.width / 2 + Math.cos(this.startAngle() + this.angle() / 2) * ((this.o() === this.index) ? 10 : 1));
 61         }).bottom(function() {
 62             return (parentDimension.height / 2 - Math.sin(this.startAngle() + this.angle() / 2) * ((this.o() === this.index) ? 10 : 1));
 63         }).event("mouseover", function() {
 64             return this.o(this.index);
 65         }).event("mouseout", function() {
 66             return this.o(-1);
 67         });
 68     };
 69 
 70     self.showLabels = function(parentDimension) {
 71         wedgeLabels = wedge.add(pv.Label).textAlign("center").textBaseline("middle").text(function() {
 72             return graphDef.data[this.index].label;
 73         });
 74         adjustLabelPosition(parentDimension, graphDef.labelFontSize);
 75         AR.Utility.setLabelProperties(graphDef, wedgeLabels, false);
 76     };
 77 
 78     self.showValues = function(parentDimension) {
 79         valueLabels = wedge.add(pv.Label).textAlign("center").textBaseline("middle").text(function() {
 80             return graphDef.data[this.index].value;
 81         });
 82         adjustLabelPosition(parentDimension, graphDef.labelFontSize, 1);
 83         AR.Utility.setLabelProperties(graphDef, valueLabels, false);
 84     };
 85 
 86     self.showLegends = function() {
 87         AR.Utility.addLegendToObject(wedge, graphDef.data, graphDef);
 88     };
 89     wedge.data(pv.normalize(dataValues));
 90     properties.forEach(function(property) {
 91         var upcasedProp = property.substring(0, 1).toUpperCase() + property.substring(1);
 92         if (graphDef["show" + upcasedProp]) {
 93             self["show" + upcasedProp](parentDimension);
 94         }
 95     });
 96     wedge.title(function() {
 97         return AR.Utility.getToolTipText(graphDef.data, this.index);
 98     });
 99     if (graphDef.presetPalette) {
100         AR.Utility.setPalette(wedge, AR.Utility.getPaletteColors(graphDef));
101     }
102     //TODO: add tool tip at the right place
103     if (graphDef.toolTip) {
104         AR.Utility.setToolTip(graphDef, wedge, "s");
105     }
106     self.adjustPosition(parentDimension);
107 };

110 /**
111  * Wedge API to construct a Pie Graph
112  * @param {object}
113  *            [graphDef] An object containing the graph properties and the data
114  * @extends AR.Graph
115  */
116 AR.PieGraph = function(graphDef) {
117     var self = this;
118     var maxValue = AR.Utility.findMaxValue(graphDef);
119     self.setVerGridShow = function(status) {
120         if (status === true) {
121             AR.Graph.prototype.setVerGridShow.apply(self, [maxValue, AR.Utility.scale.linear]);
122         }
123     };
124     self.setHorGridShow = function(status) {
125         if (status === true) {
126             AR.Graph.prototype.setHorGridShow.apply(self, [maxValue, AR.Utility.scale.linear]);
127         }
128     };
129     AR.Graph.apply(self, [graphDef]);
130     var wedges = new AR.Wedge(self._dimension, self._panel, graphDef);
131     //TODO: add other function such as changing the pallete etc  
132     // Would kinda serve as APIs for a new user who's come in..
133     self.setWidth = function(width) {
134         AR.Graph.prototype.setWidth.call(self, width);
135         wedges.adjustPosition(self._dimension);
136     };
137     self.setHeight = function(height) {
138         AR.Graph.prototype.setHeight.call(self, height);
139         wedges.adjustPosition(self._dimension);
140     };
141 };
142 AR.PieGraph.prototype = AR.extend(AR.Graph);

144 /**
145  * The basefunction for the Donut initialization.
146  * Initializes the WedgeGraph. Sets the data
147  * @params {Object}
148  *             [graphDef] graphDef object supplied by the User for which the graph needs to be plotted. It contains graph properties and the data
149  * @params {Object}
150  *             [parentDimension] parentDimension object that will indicate the dimensions of AR.Graph
151  * @params {Object}
152  *             [panel] A panel object indicating the Graph Panel in which the current Wedge graph will be displayed
153  * This is a base class and is called by different Wedge implementations (Eg : Donut or Pie)
154  */
155 AR.Donut = function(parentDimension, panel, graphDef) {
156     var properties = ["values", "labels", "legends"];
157     var wedge = panel.add(pv.Wedge);
158     var wedgeLabels, valueLabels;
159     var dataValues = AR.Utility.getDataArray(graphDef.data);
160     var self = this;
161     var adjustLabelPosition = function(parentDimension, labelFontSize, isValueLabel) {
162         var labels, shift;
163         if (isValueLabel) {
164             labels = valueLabels;
165             shift = labelFontSize;
166         } else {
167             labels = wedgeLabels;
168             shift = 0;
169         }
170         if (labels) {
171             labels.left(function() {
172                 return ((wedge.outerRadius() + wedge.innerRadius()) / 2 * Math.cos(wedge.midAngle()) + (parentDimension.width) / 2);
173             }).bottom(function() {
174                 return (-(wedge.outerRadius() + wedge.innerRadius()) / 2 * Math.sin(wedge.midAngle()) + (parentDimension.height) / 2 - shift);
175             });
176         }
177     };
178 
179     var adjustRadius = function(parentDimension) {
180         if (graphDef.outerRadius !== null) {
181             wedge.outerRadius(AR.Utility.getSize(graphDef,"wedge",graphDef.outerRadius));
182         } else {
183             wedge.outerRadius(function() {
184                 return (parentDimension.width < parentDimension.height ? (parentDimension.width - 30) / 2 : (parentDimension.height - 40) / 2);
185             });
186         }
187 
188         if (graphDef.innerRadius !== null) {
189             wedge.innerRadius(AR.Utility.getSize(graphDef,"wedge",graphDef.innerRadius));
190         } else {
191             wedge.innerRadius(function() {
192                 return (parentDimension.width < parentDimension.height ? ((parentDimension.width - 30) / 2) - 40 : ((parentDimension.height - 40) / 2)) - 40;
193             });
194         }
195     };
196 
197     var adjustAngle = function(parentDimension) {
198         wedge.angle(function(d) {
199             return ((d) * 2 * Math.PI);
200         });
201     };
202     self.showLabels = function(parentDimension) {
203         wedgeLabels = wedge.add(pv.Label).textAlign("center").textBaseline("middle").text(function() {
204             return graphDef.data[this.index].label;
205         });
206         adjustLabelPosition(parentDimension, graphDef.labelFontSize);
207         AR.Utility.setLabelProperties(graphDef, wedgeLabels, false);
208     };
209 
210     self.showValues = function(parentDimension) {
211         valueLabels = wedge.add(pv.Label).textAlign("center").textBaseline("middle").text(function() {
212             return graphDef.data[this.index].value;
213         });
214         adjustLabelPosition(parentDimension, graphDef.labelFontSize, 1);
215         AR.Utility.setLabelProperties(graphDef, valueLabels, false);
216     };
217     self.adjustPosition = function(parentDimension) {
218         adjustRadius(parentDimension);
219         adjustAngle(parentDimension);
220     };
221 
222     wedge.data(pv.normalize(dataValues));
223 
224     wedge.title(function() {
225         return AR.Utility.getToolTipText(graphDef.data, this.index);
226     });
227 
228     //TODO: add tool tip at the right place
229     if (graphDef.toolTip && graphDef.toolTip === 1) {
230         AR.Utility.setToolTip(graphDef, wedge, "s");
231     }
232     self.showLegends = function() {
233         AR.Utility.addLegendToObject(wedge, graphDef.data, graphDef);
234     };
235     self.adjustPosition(parentDimension);
236     properties.forEach(function(property) {
237         var upcasedProp = property.substring(0, 1).toUpperCase() + property.substring(1);
238         if (graphDef["show" + upcasedProp]) {
239             self["show" + upcasedProp](parentDimension);
240         }
241     });
242     if (graphDef.presetPalette) {
243         AR.Utility.setPalette(wedge, AR.Utility.getPaletteColors(graphDef));
244     }
245 };
