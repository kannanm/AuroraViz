  3 /**
  4  * The basefunction for the Line initialization.
  5  * Initializes the LineGraph. Sets the data
  6  * @params {Object}
  7  *             [graphDef] graphDef object supplied by the User for which the graph needs to be plotted. It contains graph properties and the data
  8  * @params {Object}
  9  *             [parentDimension] parentDimension object that will indicate the dimensions of AR.Graph
 10  * @params {Object}
 11  *             [panel] A panel object indicating the Graph Panel in which the current Line graph will be displayed
 12  * This is a base class and is called by different Line implementations.
 13  */
 14 AR.Line = function(parentDimension, panel, graphDef, step) {
 15     var self = this;
 16     var noOfRecords = graphDef.data.length;
 17     var line = panel.add(pv.Line);
 18     line.data(AR.Utility.getDataArray(graphDef.data));
 19     var dots = line.add(pv.Dot).lineWidth(1);
 20     dots.title(function() {
 21         return AR.Utility.getToolTipText(graphDef.data, this.index);
 22     });
 23     var properties = ["dotSize", "lineWidth", "lineColor"];
 24     self.setDotSize = function(size) {
 25         dots.shapeRadius(size);
 26     };
 27     self.setLineWidth = function(width) {
 28         line.lineWidth(width);
 29     };
 30     self.setLineColor = function(color) {
 31         line.strokeStyle(color);
 32     };
 33     var setProperties = function() {
 34         properties.forEach(function(property) {
 35             var upcasedProp = property.substring(0, 1).toUpperCase() + property.substring(1);
 36             if (graphDef[property]) {
 37                 self["set" + upcasedProp](graphDef[property]);
 38             }
 39         });
 40     };
 41     self.adjustLabel = function(parentDimension) {
 42         self._label = dots.add(pv.Label).bottom(0).text(function() {
 43             return graphDef.data[this.index].label;
 44         }).textAngle(-Math.PI / 4).textAlign("right").textBaseline("top");
 45     };
 46     self.adjustValuePositions = function(parentDimension) {
 47         dots.anchor("bottom").add(pv.Label).text(function() {
 48             return graphDef.data[this.index].value;
 49         }).font(AR.Utility.getSize(graphDef,"gridLabels",AR.constants.values.smallLabels.size) + "px Arial");
 50     };
 51     self.adjustBottom = function(parentDimension) {
 52         var pointHeight = pv.Scale.linear(0, AR.Utility.findMax(graphDef.data)).range(0, parentDimension.height - 40);
 53         line.bottom(function(d) {
 54             return pointHeight(d);
 55         });
 56     };
 57     self.adjustLeft = function(parentDimension) {
 58         line.left(function() {
 59             return (this.index * (parentDimension.width - 30) / (noOfRecords)) + 30;
 60         });
 61     };
 62     self.adjustPositions = function(parentDimension) {
 63         self.adjustBottom(parentDimension);
 64         self.adjustLeft(parentDimension);
 65     };
 66     if (graphDef.showValues) {
 67         self.adjustValuePositions(parentDimension);
 68     }
 69     AR.Utility.setToolTip(graphDef, dots, "s");
 70     if (graphDef.presetPalette) {
 71         AR.Utility.setPalette(dots, AR.Utility.getPaletteColors(graphDef));
 72     }
 73     if (graphDef.showLabels) {
 74         self.adjustLabel(parentDimension);
 75         AR.Utility.setLabelProperties(graphDef, self._label, false);
 76     }
 77     self.adjustPositions(parentDimension);
 78     setProperties();
 79     if (step === true) {
 80         line.interpolate("step-after");
 81     }
 82     if (graphDef.showLegends) {
 83         AR.Utility.addLegendToObject(dots, graphDef.data, graphDef);
 84     }
 85 };

 87 /**
 88  * The basefunction for the Multi LineGraph initialization.
 89  * Initializes the LineGraph. Sets the data
 90  * @params {Object}
 91  *             [graphDef] graphDef object supplied by the User for which the graph needs to be plotted. It contains graph properties and the data
 92  * @params {Object}
 93  *             [parentDimension] parentDimension object that will indicate the dimensions of AR.Graph
 94  * @params {Object}
 95  *             [panel] A panel object indicating the Graph Panel in which the current Line graph will be displayed
 96  * @params {Object}
 97  *             [data] The data for line chart plotting
 98  * @params {value}
 99  *             [maxValue] The value for setting the scale
100  * This is a base class and is called by different Line implementations.
101  */
102 AR.MLine = function(parentDimension, panel, graphDef, data, maxValue, step, color, seriesname, seriesNumber) {
103     var self = this;
104     var noOfRecords = data.length;
105     var line = panel.add(pv.Line);
106     line.data(AR.Utility.getDataArray(data));
107     var dots = line.add(pv.Dot).lineWidth(4).fillStyle(color);
108     dots.title(function() {
109         return AR.Utility.getToolTipText(data, this.index);
110     });
111     var properties = ["dotSize", "lineWidth", "lineColor"];
112     self.setDotSize = function(size) {
113         dots.shapeRadius(size);
114     };
115     self.setLineWidth = function(width) {
116         line.lineWidth(width);
117     };
118     self.setLineColor = function(color) {
119         line.strokeStyle(color);
120     };
121     var setProperties = function() {
122         properties.forEach(function(property) {
123             var upcasedProp = property.substring(0, 1).toUpperCase() + property.substring(1);
124             if (graphDef[property]) {
125                 self["set" + upcasedProp](graphDef[property]);
126             }
127         });
128     };
129     self.showLabels = function(parentDimension) {
130         self._label = dots.add(pv.Label).bottom(0).text(function() {
131             return graphDef.categories[0].category[this.index].label;
132         }).textBaseline("top");
133     };
134     self.adjustValuePositions = function(parentDimension) {
135         dots.anchor("bottom").add(pv.Label).text(function() {
136             return data[this.index].value;
137         }).font(AR.Utility.getSize(graphDef,"gridLabels",AR.constants.values.smallLabels.size) + "px Arial");
138     };
139     self.adjustBottom = function(parentDimension) {
140         var pointHeight = pv.Scale.linear(0, maxValue).range(0, parentDimension.height - 40);
141         line.bottom(function(d) {
142             return pointHeight(d);
143         });
144     };
145     self.adjustLeft = function(parentDimension) {
146         line.left(function() {
147             return (this.index * (parentDimension.width - 30) / (noOfRecords)) + 30;
148         });
149     };
150 
151     self.adjustPositions = function(parentDimension) {
152         self.adjustBottom(parentDimension);
153         self.adjustLeft(parentDimension);
154     };
155     AR.Utility.setToolTip(graphDef, dots, "s");
156     if (graphDef.showValues) {
157         self.adjustValuePositions(parentDimension);
158     }
159     if (graphDef.showLabels) {
160         self.showLabels(parentDimension);
161         AR.Utility.setLabelProperties(graphDef, self._label, false);
162     }
163     self.adjustPositions(parentDimension);
164     setProperties();
165     if (step === true) {
166         line.interpolate("step-after");
167     }
168     if (color) {
169         line.strokeStyle(color);
170     }
171     if (graphDef.showLegends) {
172         AR.Utility.createLegends(panel, seriesNumber, seriesname, color, graphDef);
173     }
174 };

176 /**
177  * API to construct a Line Graph
178  * @param {object}
179  *            [graphDef] An object containing the graph properties and the data
180  * @extends AR.Graph
181  */
182 AR.LineGraph = function(graphDef) {
183     var line;
184     var self = this;
185     var flag, palette;
186     var maxValue = AR.Utility.findMaxValue(graphDef);
187     self.setVerGridShow = function(status) {
188         if (status === true) {
189             AR.Graph.prototype.setVerGridShow.apply(self, [maxValue, AR.Utility.scale.linear]);
190         }
191     };
192     self.setHorGridShow = function(status) {
193         if (status === true) {
194             AR.Graph.prototype.setHorGridShow.apply(self, [maxValue, AR.Utility.scale.linear]);
195         }
196     };
197     AR.Graph.apply(self, [graphDef]);
198     if (graphDef.isLineStyleStep === true) {
199         flag = true;
200     }
201     if (graphDef.presetPalette) {
202         colors = AR.Utility.getPaletteColors(graphDef);
203     }
204     if (graphDef.dataset) {
205 
206         var dataset = graphDef.dataset;
207         for (i = 0; i < graphDef.dataset.length; i++) {
208             var noOfRecords = dataset.length * dataset[i].data.length;
209             panel = self._panel.add(pv.Panel).left((self._dimension.width - 30) / (noOfRecords));
210             line = new AR.MLine(self._dimension, panel, graphDef, dataset[i].data, maxValue, flag, colors[i % colors.length], dataset[i].seriesname, i);
211         }
212     } else {
213         line = new AR.Line(self._dimension, self._panel, graphDef, flag);
214     }
215 
216     self.setWidth = function(width) {
217         AR.Graph.prototype.setWidth.call(self, width);
218         line.adjustPositions(self._dimension);
219     };
220     self.setHeight = function(height) {
221         AR.Graph.prototype.setHeight.call(self, height);
222         line.adjustPositions(self._dimension);
223         self.setHorRules(AR.Utility.findMax(graphDef.data), AR.Utility.scale.linear);
224     };
225 };
226 AR.LineGraph.prototype = AR.extend(AR.Graph);
