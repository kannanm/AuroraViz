1 /**
  2  * @Description The top level namespace for all the utility functions in Aurora.
  3  * @namespace The top-level utility namespace, <tt>AR.Utility</tt>.
  4  */
  5 AR.Utility = {};
  6 
  7 /**
  8  * @class It is a Utilty Class used to look the data as a table with columns and rows. 
  9  * It also provides functions to access data.
 10  * 
 11  * @param {Array}
 12  *            columnNames It is an array consisting of the names of the columns
 13  * @param {Array}
 14  *            data It is an array containing the data for the table. It can either be an array of arrays or array of objects
 15  * @param {boolean}
 16  * 			  isJson a boolean field representing if the data is array of arrays or array of Objects
 17  */
 18 
 19 AR.Utility.Table = function (columnNames, data, isJson) {
 20  this.columnNames = columnNames;
 21  this._data = data;
 22  this._isAA = ! isJson;
 23 };
 24 
 25 /**
 26  * Returns the data value based on the rowIndex and the column name
 27  * @param {Number} rowIndex
 28  *             This is the index of the row from where the data is to be accessed
 29  * @param {String} columnName
 30  * 				This is the name of the column from where the data is to be accessed
 31  * @returns {Number} The data value
 32  */
 33 
 34 AR.Utility.Table.prototype.dataValue = function (rowIndex, columnName) {
 35  var row = this._data[rowIndex];
 36  if(this._isAA){
 37   var colIndex = this.columnNames.indexOf(columnName);
 38   return row[colIndex];
 39  }else{
 40   return row[columnName];
 41  }
 42 };
 43 
 44 /**
 45  * Returns the data value based on the rowIndex and the column index
 46  * @param {Number} rowIndex
 47  *             This is the index of the row from where the data is to be accessed
 48  * @param {Number} colIndex
 49  * 				This is the index of the column from where the data is to be accessed
 50  * @returns {Number} The data value
 51  */
 52 
 53 AR.Utility.Table.prototype.dataValueByRowAndColIndex = function (rowIndex, colIndex) {
 54  if(this._isAA){
 55   return this._data[rowIndex][colIndex];
 56  }else{
 57   return this._data[rowIndex][this.columnNames[colIndex]];
 58  }
 59 
 60 };
 61 
 62 /**
 63  * Returns the row based on the row index
 64  * @param {Number} rowIndex
 65  *             This is the index of the row 
 66  * @returns {Array} The row
 67  */
 68 AR.Utility.Table.prototype.row = function (rowIndex) {
 69  return this._data[rowIndex];
 70 };
 71 
 72 /**
 73  * Returns size of the data
 74  * @returns {Number} The size of the data
 75  * 
 76  */
 77 AR.Utility.Table.prototype.getLength = function () {
 78  return this._data.length;
 79 };

 81 /**
 82  * Checks if the <tt>chartType</tt> exists.
 83  * @param {string} chartType 
 84  * 			It is the type of the Chart that the user wants to initialize
 85  * @return {boolean} true if the <tt>chartType</tt> exists, otherwise false
 86  */
 87 
 88 AR.Utility.checkChartType = function(chartType){
 89 	return chartType?true:false;
 90 };
 91 
 92 /**
 93  * Checks if the <tt>graphData</tt> contains both the data and the series.
 94  * @param {json} graphData 
 95  * 			The data for the graph
 96  * @return {boolean} true if the <tt>graphData</tt> is valid, otherwise false
 97  */
 98 
 99 AR.Utility.checkGraphData = function(graphData){
100 	var isValid = true;
101 	isValid = isValid && (graphData.data)? true: false;
102 	isValid = isValid && (graphData.series)? true: false; 
103 	return isValid;
104 };
105 
106 /**
107  * Checks if <tt>chartType</tt> exists and the data and the series name are present and their length is equal 
108  * as with every series a data is associated  
109  * @param {string} chartType 
110  * 		 	It is the type of the Chart that the user wants to initialize
111  * @param {json} graphData
112  * 			It is an object containing the data and the series name for the graph
113  * 
114  */
115 AR.Utility.checkParams = function(chartType,graphData){
116 	var isValid = true;
117 	isValid = isValid && AR.Utility.checkChartType(chartType);
118 	isValid = isValid && AR.Utility.checkGraphData(graphData);
119 	return isValid;
120 };
