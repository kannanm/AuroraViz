  4 /**
  5  * A constructor function to create a stacked area graph.
  6  *
  7  * @class
  8  * @param {object}
  9  *            [graphDef] An object containing the graph properties and the data
 10  * @extends AR.Graph
 11  */
 12 AR.StackedAreaGraph = function(graphDef) {
 13 
 14     var self = this;
 15     var dataset = graphDef.dataset;
 16     var colors = AR.Utility.getPaletteColors(graphDef);
 17     /**
 18      * @private
 19      * It is sum of maximum values from each categories because the scale will depend on it as it is a stacked graph
 20      */
 21     var sumOfMaxValues = (function() {
 22         var sum = 0,
 23             i;
 24         for (i = 0; i < dataset.length; i++) {
 25             sum += AR.Utility.findMax(dataset[i].data);
 26         }
 27         return sum;
 28     }());
 29 
 30     /**
 31      * @private
 32      * Total number of records in the data set
 33      */
 34     var noOfRecords = (function() {
 35         var records = 0;
 36         for (i = 0; i < dataset.length; i++) {
 37             if (dataset[i].data.length > records) {
 38                 records = dataset[i].data.length;
 39             }
 40         }
 41         return records;
 42     }());
 43 
 44     /**
 45      * It is an array of arrays. The inner array representing the data in each category.
 46      */
 47     var dataArray = (function() {
 48         var arr = [],
 49             innerArr, i, j, obj;
 50         for (i = 0; i < dataset.length; i++) {
 51             innerArr = [];
 52             var data = dataset[i].data;
 53             for (j = 0; j < data.length; j++) {
 54                 obj = {};
 55                 obj.x = j;
 56                 obj.y = data[j].value;
 57                 innerArr.push(obj);
 58             }
 59             arr.push(innerArr);
 60         }
 61         return arr;
 62     }());
 63 
 64     self.setVerGridShow = function(status) {
 65         if (status === true) {
 66             AR.Graph.prototype.setVerGridShow.apply(self, [dataset.length, AR.Utility.scale.linear]);
 67         }
 68     };
 69     self.setHorGridShow = function(status) {
 70         if (status === true) {
 71             AR.Graph.prototype.setHorGridShow.apply(self, [sumOfMaxValues, AR.Utility.scale.linear]);
 72         }
 73     };
 74     AR.Graph.apply(self, [graphDef]);
 75 
 76     var xScale = pv.Scale.linear(0, noOfRecords).range(0, self._dimension.width - 40),
 77         yScale = pv.Scale.linear(0, sumOfMaxValues).range(0, self._dimension.height - 40);
 78 
 79 /*
 80      * The function creates the stack area Object
 81      */
 82     var createStackedAreaGraph = function() {
 83         self._areaObj = self._panel.add(pv.Layout.Stack).layers(dataArray).x(function(d) {
 84             return xScale(d.x) + AR.AreaConstants.xOffset;
 85         }).y(function(d) {
 86             return yScale(d.y);
 87         }).layer.add(pv.Area);
 88         self._areaObj.strokeStyle(function() {
 89             return colors[this.parent.index % colors.length];
 90         }).fillStyle(function() {
 91             return this.strokeStyle().alpha(0.2);
 92         });
 93         self._areaObj.segmented(true);
 94 
 95     };
 96     createStackedAreaGraph();
 97     self._areaObj.title(function() {
 98         return (graphDef.dataset[this.parent.index].seriesname);
 99     });
100     if (graphDef.showValues) {
101         self._areaObj.anchor("top").add(pv.Label).text(function(d) {
102             var dataObj = dataArray[this.parent.index];
103             return (dataObj[this.index].y);
104         }).textBaseline("bottom")
105         .font(AR.Utility.getSize(graphDef,"gridLabels",AR.constants.values.smallLabels.size) + "px Arial");
106     }
107     if (graphDef.showLabels) {
108         var labels = self._areaObj.add(pv.Label).text(function() {
109             return graphDef.categories[0].category[this.index].label;
110         }).bottom(0).textBaseline("top");
111         AR.Utility.setLabelProperties(graphDef, labels, false);
112     }
113     if (graphDef.showLegends) {
114         var i = 0;
115         for (i = 0; i < dataset.length; i++) {
116             AR.Utility.createLegends(self._panel, i, dataset[i].seriesname, colors[i % dataset.length],graphDef);
117         }
118     }
119     if (graphDef.interpolated) {
120         self._areaObj.interpolate("step-after");
121     }
122     AR.Utility.setToolTip(graphDef, self._areaObj, "s");
123 };
124 AR.StackedAreaGraph.prototype = AR.extend(AR.Graph);

126 /**
127  * A constructor function to create a Area Object.
128  *
129  * @class
130  * @param {object}
131  *            [parentDimension] The dimension Object of the parent element. Here the parent element is the panel
132  * @param {object}
133  *            [panel] The panel Object to which the graph objects will be added
134  * @param {object}
135  *            [graphDef] An object containing the graph properties and the data
136  * @param {object}
137  *            [dataObj] An object containing the graph data
138  * @param {object}
139  *            [color] If defined then the fill color of area object will be set to color
140  * @param {object}
141  *            [seriesname] If given defined then this area object is a part for multi series area graph and series name is <b>seriesname</b>
142  *
143  * @author Aditya Gaur
144  */

146 AR.Area = function(parentDimension, panel, graphDef, dataObj, color, seriesname, seriesNumber) {
147     var self = this;
148     self._areaObj = panel.add(pv.Area);
149     var dataArr = AR.Utility.getDataArray(dataObj);
150     self._areaObj.data(dataArr);
151 /*
152      * Sets the height of the area object
153      */
154     var setHeight = function() {
155         var maxVal = AR.Utility.findArrayMax(dataArr);
156         var xScale = pv.Scale.linear(0, maxVal).range(0, parentDimension.height - 40);
157         self._areaObj.height(function(d) {
158             return xScale(d);
159         });
160     };
161 
162 /*
163      *
164      * Sets the left property of the area object
165      */
166     var setLeft = function() {
167         var noOfRecords = dataArr.length;
168         var xScale = pv.Scale.linear(0, noOfRecords).range(0, parentDimension.width - 40);
169         self._areaObj.left(function(d) {
170             return xScale(this.index) + AR.AreaConstants.xOffset;
171         });
172     };
173 
174 /*
175      * Sets the bottom property of the area object
176      */
177     var setBottom = function() {
178         self._areaObj.bottom(0);
179     };
180 
181 /*
182      * Sets the fill color of the area object if <b>color</b> is defined then set to <b>color</b>
183      */
184     var setFillStyle = function() {
185         self._areaObj.strokeStyle(color || graphDef.areaColor);
186         self._areaObj.fillStyle(function() {
187             return this.strokeStyle().alpha(0.2);
188         });
189     };
190 
191 
192     self.setAreaAttributes = function() {
193         var self = this;
194         setBottom();
195         setHeight();
196         setLeft();
197         setFillStyle();
198         self._areaObj.segmented(true);
199         if (graphDef.interpolated === true) {
200             self._areaObj.interpolate("step-after");
201         }
202     };
203     self._areaObj.title(function() {
204         return (seriesname ? seriesname : dataObj[this.index].toolTipText ? dataObj[this.index].toolTipText : dataObj[this.index].value);
205     });
206     if (graphDef.showLabels) {
207         var label = self._areaObj.add(pv.Label).textBaseline("top").text(function() {
208             return seriesname ? graphDef.categories[0].category[this.index].label : dataObj[this.index].label;
209         });
210         AR.Utility.setLabelProperties(graphDef, label, false);
211     }
212     if (graphDef.showValues) {
213         var values = self._areaObj.anchor("top").add(pv.Label).textBaseline("bottom").text(function() {
214             return dataObj[this.index].value;
215         }).font(AR.Utility.getSize(graphDef,"gridLabels",AR.constants.values.smallLabels.size) + "px Arial");
216     }
217     AR.Utility.setToolTip(graphDef, self._areaObj, "s");
218     self.setAreaAttributes();
219 
220     //Note: Legends for a single series area graph does no make sense. Thus not implemented
221     if (graphDef.showLegends) {
222         AR.Utility.createLegends(panel, seriesNumber, seriesname, color, graphDef);
223     }
224 };

228     var self = this;
229     var area, colors;
230     var maxVal = AR.Utility.findMaxValue(graphDef);
231     var dataset = graphDef.dataset;

233         if (status === true) {
234             AR.Graph.prototype.setVerGridShow.apply(self, [maxVal, AR.Utility.scale.linear]);
235         }
236     };
237     self.setHorGridShow = function(status) {
238         if (status === true) {
239             AR.Graph.prototype.setHorGridShow.apply(self, [maxVal, AR.Utility.scale.linear]);
240         }
241     };
242     AR.Graph.apply(self, [graphDef]);
243     if (graphDef.presetPalette) {
244         colors = AR.Utility.getPaletteColors(graphDef);
245     }
246     if (dataset) {
247         for (i = 0; i < dataset.length; i++) {
248             var panel = self._panel.add(pv.Panel);
249             bar = new AR.Area(self._dimension, panel, graphDef, dataset[i].data, colors[i % colors.length], graphDef.dataset[i].seriesname, i);
250         }
251     }
252     else {
253         area = new AR.Area(self._dimension, self._panel, graphDef, graphDef.data);
254     }
255
