  3 Array.prototype.max = Array.prototype.max ||
  4 function() {
  5     var max = this[0];
  6     var len = this.length;
  7     var i;
  8     for (i = 1; i < len; i++) {
  9         if (this[i] > max) {
 10             max = this[i];
 11         }
 12     }
 13     return max;
 14 };

 24 /**
 25  
 26  * Returns a prototype object suitable for extending the
 27  *         given class <tt>f</tt>. For more details, see Douglas Crockford's
 28  *         essay on prototypal inheritance.
 29  * @param  {function}
 30  *            f a constructor.
 31  * @returns a suitable prototype object.
 32  * @see Douglas Crockford's essay on <a
 33  *      href="http://javascript.crockford.com/prototypal.html">prototypal
 34  *      inheritance</a>.
 35  */
 36 AR.extend = function(f) {
 37     function g() {}
 38     g.prototype = f.prototype || f;
 39     return new g();
 40 };

 42 /**
 43  
 44  * Adds the javascript file specified by the <tt>jsPath</tt>,
 45  *         at the position <tt>pos</tt>.
 46  * @param {string}
 47  *            [jsPath] the path of the JS file to be included.
 48  * @param {string}
 49  *            [pos](optional)the dom element in which the script tag to include
 50  *            js file will appear. If unspecified the js file will be included
 51  *            in head.
 52  */
 53 AR.addJavascript = function(jsPath, pos, dataPath, callback) {
 54     var position = pos || "head";
 55     var posElem = document.getElementsByTagName(position)[0];
 56     var scriptTag = document.createElement('script');
 57     scriptTag.setAttribute('type', 'text/javascript');
 58     scriptTag.setAttribute('src', jsPath);
 59     if (dataPath) {
 60         scriptTag.setAttribute('data-path', dataPath);
 61     }
 62     posElem.appendChild(scriptTag);
 63     return scriptTag;
 64 };

 84     cssTag.setAttribute('rel', 'stylesheet');

 88 /**
 89  * @class
 90  * The basefunction for the graph initilization.
 91  * Initializes the panel and sets parentDimensions and property values
 92  * @param {json}
 93  *             graphDef is the current graph Dimension
 94  * This method should be the first method that should be called when creating any kind of graph.
 95  * Creates the panel, sets its dimensions, labels and property values and returns.
 96  */
 97 AR.Graph = function(graphDef) {
 98     var self = this;
 99     var upcasedProp;
100     self.__graphDef = graphDef;
101     self._pos = undefined;
102     self._dimension = {
103         height: 400,
104         width: 400,
105         left: 60,
106         right: 20,
107         bottom: 60,
108         top: 60
109     };
110     self._caption = undefined;
111     self._horAxisLabel = undefined;
112     self._verAxisLabel = undefined;
113     self._horGrid = undefined;
114     self._verGrid = undefined;
115     self._outerPanel = new pv.Panel().left(0).right(0).bottom(0).top(0);
116     self._panel = self._outerPanel.add(pv.Panel);
117     self._panel.left(self._dimension.left).right(self._dimension.right).bottom(self._dimension.bottom).top(self._dimension.top);
118     self._horAxis = self._panel.add(pv.Rule).bottom(0);
119     self._verAxis = self._panel.add(pv.Rule).left(0);
120     this.properties.forEach(function(property) {
121         upcasedProp = property.substring(0, 1).toUpperCase() + property.substring(1);
122         if (graphDef[property]) {
123             self["set" + upcasedProp](graphDef[property]);
124         }
125     });
126     var styles = graphDef.style;
127     var prop;
128     for (prop in styles) {
129         upcasedProp = prop.substring(0, 1).toUpperCase() + prop.substring(1);
130         if (styles.hasOwnProperty(prop)) {
131             self["set" + upcasedProp](styles[prop]);
132         }
133     }
134 };

290  * Sets a graph's caption.
291  * Caption is one of the properties of any graph
292  * @param {string}
293  *             Sets the caption for the given graph as the supplied parameter value
294  */
295 AR.Graph.prototype.setCaption = function(newVal) {
296     var self = this;
297     if (self._caption) {
298         self._caption.text(newVal);
299     } else {
300         self._caption = self._outerPanel.add(pv.Label).text(newVal).left(self._dimension.width / 2 + 40).top(0).textAlign("center").textBaseline("top");
301     }
302 };

304 /**
305  * Sets a graph's width.
306  * Width is one of the properties of a graph
307  * @param {string}
308  *             [width] Sets the width for the given graph as the supplied parameter value
309  */
310 AR.Graph.prototype.setWidth = function(width) {
311     var self = this;
312     self._dimension.width = width - self._dimension.left - self._dimension.right;
313     self._outerPanel.width(width);
314     if (self._caption) {
315         self._caption.left(self._dimension.width / 2);
316     }
317 };

358 /**
359  * Sets horizontal rules in a graph.
360  * Rules are nothing but lines that divide the graph to form a grid like structure
361  * @param {integer}
362  *               [maxValue] This is the maximimum value that the graph would have in horizontal direction. It is required to calculate the scale.
363  * @param {object}
364  *            [scaleType] It takes the scaleType (Linear/Ordinal etc)
365  */
366 
367 AR.Graph.prototype.setHorGridShow = function(maxValue, scaleType) {
368     var self = this;
369     self._y = pv.Scale[scaleType](0, maxValue).range(0, self._dimension.height - 40);
370     if (!self._horGrid) {
371         self._horGrid = self._panel.add(pv.Rule);
372         self._horGrid.data(self._y.ticks().splice(1, self._y.ticks().length)).bottom(self._y);
373     } else {
374         self._horGrid.bottom(self._y);
375     }
376 };
