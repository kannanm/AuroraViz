 18 /**
 19  * Creates the JSON <tt>graphDef</tt> for the chart. The aurora source uses
 20  * <tt>graphDef</tt> to generate chart
 21  * @param chartType
 22  * @returns {Object} graphDef
 23  */
 24 
 25 ARV.CG.getData = function(chartType) {
 26     var graphDef = {},
 27         prop;
 28     graphDef.style = ARV.styles;
 29     for (prop in ARV.commonProperties) {
 30         graphDef[prop] = ARV.commonProperties[prop];
 31     }
 32     graphDef.data = ARV.defaultData[chartType];
 33     if (chartType === "MultiBarGraph") {
 34         graphDef.categories = ARV.defaultCategories;
 35         graphDef.dataset = ARV.defaultDataSet;
 36     }
 37     return graphDef;
 38 };

 40 /**
 41  * The function gets the style property value from the front end and sets these style properties in graphDef
 42  */
 43 ARV.CG.updateStyleProperties = function() {
 44     var styles = ARV.graphDef.style,
 45         prop;
 46     for (prop in styles) {
 47         var elem = $("#" + prop);
 48         var value = elem.val();
 49         if (prop === "verGridShow" || prop === "horGridShow" || prop === "verGridLabelShow" || prop === "horGridLabelShow") {
 50             value = elem.attr("checked") ? true : false;
 51         }
 52         if (styles.hasOwnProperty(prop)) {
 53             styles[prop] = value;
 54         }
 55     }
 56 };
 57 
 58 /**
 59  * The function sets the properties of chart in the graphDef
 60  */
 61 ARV.CG.updateGraphProperties = function() {
 62     var prop;
 63     for (prop in ARV.graphDef) {
 64         var elem = $("#" + prop);
 65         var value = elem.val();
 66         if (prop === "toolTip" || prop === "showLabels" || prop === "showValues" || prop === "isLineStyleStep" || prop === "interpolated" || prop === "segmented" || prop === "showLegends") {
 67             value = elem.attr("checked") ? true : false;
 68         }
 69         if (ARV.graphDef.hasOwnProperty(prop) && value !== undefined) {
 70             ARV.graphDef[prop] = value;
 71         }
 72     }
 73 };

 75 /**
 76  * The function checks if the palette has changed and if the palette has changed, it creates
 77  * an array of colors and sets it in <tt>graphDef</tt>
 78  */
 79 ARV.CG.checkPaletteChanged = function() {
 80 	var generatorNS = ARV.CG;
 81     if (generatorNS.paletteChanged) {
 82         var colors = [];
 83         $("#colors input").each(function() {
 84             colors.push($(this).val());
 85         });
 86         generatorNS.graphDef.presetPalette = "custom";
 87         generatorNS.graphDef.paletteColors = colors;
 88     }
 89 };

 95 ARV.CG.getNumberOfSelectedMeasureAxis = function() {

101 /**
102  * The function returns the graph type selected
103  * @returns {String} The graph type
104  */
105 ARV.CG.getGraphType = function() {
106     var graphType = $("#chartTypes option:selected").attr("value");
107     var graphName = graphType + "Graph";
108     var noOfSelected = ARV.CG.getNumberOfSelectedMeasureAxis();
109     if (noOfSelected > 1) {
110         graphName = ARV.singleToMultiMap[graphName];
111     }
112     return graphName;
113 };

115 /**
116  * This function updates the JSON for the graph whenever the user makes changes on the front end
117  * this function is called.
118  * @param {string} div
119  *             The div id in which the chart is to be populated
120  */
121 ARV.CG.modifyJSON = function(div) {
122 	var generatorNS = ARV.CG;
123     var index, param, value;
124     var graphName = generatorNS.getGraphType();
125     ARV.graphDef = generatorNS.getData(ARV.JsonMap[graphName]);
126     generatorNS.updateStyleProperties();
127     generatorNS.updateGraphProperties();
128     generatorNS.checkPaletteChanged();
129     generatorNS.showGraph(graphName, div);
130 };

132 /**
133  * The function creates an instance of the <tt>graphType</tt> and renders it to the <tt>div</tt>
134  * @param graphType {String} The graph type
135  * @param div {Optional} The div id in which the chart is to be rendered. If undefined it is rendered to the div with id "chart"
136  * @param {Optional} graphDef This is the JSON that is passed to the Aurora source. If undefined then <tt>ARV.graphDef</tt> is taken.
137  */
138 ARV.CG.showGraph = function(graphType, div, graphDef) {
139     $("#chart").html("<img src='css/images/loading.gif' class='loading'/>");
140     graph = new AR[ARV.ChartTypeMap[graphType]](graphDef || ARV.graphDef);
141     $("#chart").html("");
142     graph.render(div || "chart");
143 };

145 /**
146  * Adds the colors to the according to the palette chosen to the palette tab. The user can
147  * add or remove colors from the palette
148  */
149 ARV.CG.addColorsDiv = function() {
150 	var generatorNS = ARV.CG;
151     generatorNS.paletteChanged = false;
152     var selectedVal = $("#presetPalette option:selected").attr("value");
153     var colors = AR.Utility.palettes[selectedVal];
154     var colorsDiv = $("#colors");
155     colorsDiv.html("");
156     for (generatorNS.paletteColorCounter = 0; generatorNS.paletteColorCounter < colors.length; generatorNS.paletteColorCounter++) {
157         colorsDiv.append('<input value="' + colors[generatorNS.paletteColorCounter] + '" class="colorInput colorPicker" size="8" type="text" id="color' + generatorNS.paletteColorCounter + '"/>');
158         colorsDiv.append('<button id="color' + generatorNS.paletteColorCounter + '"  class="delColorBtn">x</button>');
159     }
160     colorsDiv.append('<button id="addColorBtn" class="ui-state-default ui-corner-all"><span class="ui-icon ui-icon-circle-plus"></span></button>');
161 
162     ARV.Utility.initializeColorPicker(ARV.Utility.colorPickerCB.chartGenerator);
163 };

176 /**
177  * The function display/hides the category/measure axis divs according to the chart type selected
178  * @param isMap
179  */
180 ARV.CG.toggleParameters = function(isMap) {
181     if (isMap) {
182         $("#MapAxisSelector").css("display", "inline");
183         $("#defaultAxisSelector").css("display", "none");
184     } else {
185         $("#MapAxisSelector").css("display", "none");
186         $("#defaultAxisSelector").css("display", "inline");
187     }
188 };
189 /**
190  *
191  * @param chartType
192  */
193 ARV.CG.performActionsBasedOnChart = function(chartType) {
194 	var generatorNS = ARV.CG;
195     switch (chartType) {
196     case "table":
197         generatorNS.refreshTable();
198         generatorNS.toggleParameters(false);
199         break;
200     case "select":
201         $("#dataDiv").html("Please select a chart");
202         generatorNS.toggleParameters(false);
203         break;
204     case "Map":
205         generatorNS.toggleParameters(true);
206         $("#dataDiv").html(ARV.con[chartType + "ChartOptions"]);
207         break;
208     default:
209         $("#dataDiv").html(ARV.con[chartType + "ChartOptions"]);
210         generatorNS.toggleParameters(false);
211     }
212 };
213 /**
214  * The function adds event listeners to the different dom objects
215  */
216 ARV.CG.addEventListeners = function() {
217 	var generatorNS = ARV.CG;
218     // When any delete color button is clicked. Delete the color from the palette
219     $(".delColorBtn").live("click", function() {
220         var izzy = $(this).prev();
221         izzy.prev().remove();
222         izzy.remove();
223         $(this).remove();
224     });
225     //Whenever we click any where in the colors div the palette is changed
226     $("#colors").live("click", function() {
227         generatorNS.paletteChanged = true;
228         generatorNS.refreshGraph();
229     });
230     //Whenever add Color button is clicked a color is added to the palette with default color #000000
231     $("#addColorBtn").live("click", function() {
232         $(this).before('<input value="#000000" class="colorInput colorPicker" size="8" type="text" id="color' + generatorNS.paletteColorCounter + '"/>');
233         $(this).before('<button id="delColor' + (generatorNS.paletteColorCounter) + '"  class="delColorBtn">x</button>');
234         $("#delColor" + (generatorNS.paletteColorCounter)).button();
235         generatorNS.paletteColorCounter++;
236         ARV.Utility.initializeColorPicker(ARV.Utility.colorPickerCB.chartGenerator);
237     });
238 
239     //Whenever we change the option in the presetPalette The colors are loaded accordingly
240     $("#presetPalette option").click(function() {
241         generatorNS.addColorsDiv();
242         generatorNS.refreshGraph();
243     });
244 
245     //Use jquery ui for all the buttons
246     $("button").button();
247 
248     $("#generate").click(function(event) {
249         $(".tipsy").remove();
250         generatorNS.modifyJSON();
251     });
252 
253     //Perform actions when chart type changes
254     $("#chartTypes option").click(function() {
255         generatorNS.reset();
256         var selectedVal = $("#chartTypes option:selected").attr("value");
257         generatorNS.performActionsBasedOnChart(selectedVal);
258         ARV.Utility.createSliders(ARV.Utility.createSlidersCB.chartGenerator);
259         ARV.Utility.initializeColorPicker(ARV.Utility.colorPickerCB.chartGenerator);
260 
261     });
262 
263     //Whenever any options changed then refresh the graph
264     $("#accordion").change(function() {
265         generatorNS.refreshGraph();
266     });
267     $("#accordion input").keyup(function() {
268         generatorNS.refreshGraph();
269     });
270 };
271 /**
272  * When the options are changed and the chart type selected is a table, then the function is called
273  * Creates the slick grid table with the changed options
274  */
275 ARV.CG.refreshTable = function() {
276     var table = $("<div/>").attr("id", "tableDiv");
277     var title = $("<label/>").html($("#caption").val()).addClass("tableHeading");
278     var div = $("<div/>").attr("id", "tableCont");
279     $("#chart").html("");
280     table.append(title);
281     table.append(div);
282     $("#chart").append(table);
283     ARV.CG.grid = new Slick.Grid("#tableCont", ARV.DataLoader.TableData, ARV.DataLoader.TableColumns, ARV.DataLoader.gridOptions);
284 };

290     if (selectedVal !== "table") {
291         ARV.CG.modifyJSON();
292     }

296 /**
297  * The initialize function of the chart generator. Populates the palette tab,
298  * Makes jquery ui sliders, adds event listeners, triggers event to render chart
299  */
300 ARV.CG.initChartGenerator = function() {
301     var palette;
302     for (palette in AR.Utility.palettes) {
303         var option = $('<option />').val(palette).append(palette);
304         if (palette === "Water") {
305             option.attr("selected", "selected");
306         }
307         $('#presetPalette').append(option);
308     }
309     ARV.Utility.createSliders(ARV.Utility.createSlidersCB.chartGenerator);
310     ARV.CG.addColorsDiv();
311     ARV.CG.addEventListeners();
312     $("#measureAxisList").chosen().trigger("change");
313 };
