 29 ARV.createSliders = function(cb) {
 30     $(".slider").each(function() {
 31         var sliderID = $(this).attr("id");
 32         var textID = sliderID.substr(0, sliderID.indexOf("Slider"));
 33         var min = 0,
 34             max = 40;
 35         if (sliderID.indexOf("Rotate") >= 0) {
 36             min = -90;
 37             max = 90;
 38         }
 39         $(this).slider({
 40             range: "max",
 41             min: min,
 42             max: max,
 43             value: 2,
 44             slide:cb
 45         });
 46         $(this).slider("value", $("#" + textID).val()); /* $( "#"+textID).val($(this).slider( "value" ) ); */
 47     });
 48 };

 50 ARV.rgb2hex=  function(rgb) {
 51     rgb = rgb.match(/^rgb\((\d+),\s*(\d+),\s*(\d+)\)$/);
 52     function hex(x) {
 53         return ("0" + parseInt(x).toString(16)).slice(-2);
 54     }
 55     return "#" + hex(rgb[1]) + hex(rgb[2]) + hex(rgb[3]);
 56 };
 57 ARV.matrixToDegree = function(matrix){
 58 	var pat = /^matrix\((-*\d*.\d*),\s*(-*\d*.\d*),\s*(-*\d*.\d*),\s*(-*\d*.\d*),\s*(-*\d*.\d*)px,\s*(-*\d*.\d*)px\)/;
 59 	matrix = matrix.match(pat);
 60 	if(matrix === null ){
 61 		return 0;
 62 	}
 63 	var sign = matrix[2]!=0 ? matrix[2]/Math.abs(matrix[2]) : 0;
 64 	var degree = Math.acos(matrix[1])*180/Math.PI
 65 	return (sign*degree).toFixed(0);
 66 };
