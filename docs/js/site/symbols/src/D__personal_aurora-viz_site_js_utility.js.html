 45 /**
 46  * Create sliders
 47  * @param cb {function} The call back function for the slide event
 48  */
 49 ARV.Utility.createSliders = function(cb) {
 50     $(".slider").each(function() {
 51         var sliderID = $(this).attr("id");
 52         var textID = sliderID.substr(0, sliderID.indexOf("Slider"));
 53         var min = 0,
 54             max = 40;
 55         if (sliderID.indexOf("Rotate") >= 0) {
 56             min = -90;
 57             max = 90;
 58         }
 59         $(this).slider({
 60             range: "max",
 61             min: min,
 62             max: max,
 63             value: 2,
 64             slide: cb
 65         });
 66         $(this).slider("value", $("#" + textID).val()); /* $( "#"+textID).val($(this).slider( "value" ) ); */
 67     });
 68 };

 70 /**
 71  * Convert the rgb color value to hexadecimal value
 72  * @param rgb {String} The rgb color value e.g. rgb(0,0,0)
 73  * @returns {String} Returns the hexadecimal notation of the color e.g. #000000
 74  */
 75 
 76 ARV.Utility.rgb2hex = function(rgb) {
 77     rgb = rgb.match(/^rgb\((\d+),\s*(\d+),\s*(\d+)\)$/);
 78 
 79     function hex(x) {
 80         return ("0" + parseInt(x, 10).toString(16)).slice(-2);
 81     }
 82     return "#" + hex(rgb[1]) + hex(rgb[2]) + hex(rgb[3]);
 83 };

 85 /**
 86  * Convert the matrix angle notation to the degree notation
 87  * @param matrix {String} The matrix angle notation e.g. matrix(0, 1, -1, 0, 0px, 0px)
 88  * @returns The angle for the <tt>matrix</tt> notation e.g. 90
 89  */
 90 ARV.Utility.matrixToDegree = function(matrix) {
 91     var pat = /^matrix\((-*\d*.\d*),\s*(-*\d*.\d*),\s*(-*\d*.\d*),\s*(-*\d*.\d*),\s*(-*\d*.\d*)px,\s*(-*\d*.\d*)px\)/;
 92     matrix = matrix.match(pat);
 93     if (matrix === null) {
 94         return 0;
 95     }
 96     var sign = matrix[2] !== 0 ? matrix[2] / Math.abs(matrix[2]) : 0;
 97     var degree = Math.acos(matrix[1]) * 180 / Math.PI;
 98     return (sign * degree).toFixed(0);
 99 };
