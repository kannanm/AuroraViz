 68 /**
 69  * Populate the <tt>ARV.TableData</tt> and <tt>ARV.TableColumns</tt> from the filtered data
 70  * <tt>ARV.TableData</tt> and <tt>ARV.TableColumns</tt> will be used to create the SlickGrid
 71  * datatable
 72  */
 73 ARV.createDataForTable = function() {
 74     var i, label;
 75     var categoryLabel = $("#categoryAxisList option:selected").val();
 76     var measureLabels = $("#measureAxisList option:selected");
 77     var columns = ARV.filteredData.columns;
 78     ARV.TableColumns = [];
 79     ARV.TableData = [];
 80     for (i = 0; i < ARV.filteredData.data.length; i++) {
 81         ARV.TableData.push(ARV.filteredData.data[i]);
 82     }
 83     ARV.TableColumns.push(jlinq.from(columns).equals("field", categoryLabel).select()[0]);
 84     for (i = 0; i < measureLabels.length; i++) {
 85         label = $(measureLabels[i]).val();
 86         ARV.TableColumns.push(jlinq.from(columns).equals("field", label).select()[0]);
 87     }
 88 };

118 /**
119  * Update the data for the single series graph. Example data format for single series graph:
120  * <pre>"data": [{
121         "value": 22,
122         "label": "LG"
123         },
124     	{
125         "value": 12,
126         "label": "Samsung"
127         }
128        ]</pre>

130 ARV.modifyDataForSingleSeries = function() {
131     var dataArr = [];
132     var selected = $("#measureAxisList option:selected");
133     var field = $(selected[0]).val();
134     labelField = $("#categoryAxisList option:selected").val();
135     var data = ARV.filteredData.data;
136     for (i = 0; i < data.length; i++) {
137         var dataElement = data[i];
138         var value = parseInt(dataElement[field], 10);
139         if (isNaN(value)) {
140             $("#dialog-illegal-data").dialog("open");
141             return;
142         }
143         label = dataElement[labelField];
144         obj = {
145             value: value,
146             label: label
147         };
148         dataArr.push(obj);
149     }
150     ARV.defaultData.BarGraph = dataArr;
151 };

299 /**
300  * Updates the data for the map. Takes the latitude, longitude,size and category measures 
301  * and creates a JSON for the map. 
302  * e.g. JSON for map <pre>
303  * "data": [{ 
304         "lat": 22,
305         "lon": 22,
306         "size": 10,
307         "category": "LG"
308         },
309     	{
310         "lat": 12,
311         "lon": 12,
312         "size": 20,
313         "category": "Samsung"
314         }
315     ]</pre>
316  */
317 ARV.updateDataForMap = function(){
318 	var dataArr = [],
319  		obj ={},
320  		index = 0,
321  		xValue,	yValue, zValue,
322  		dataItem,
323  		lat = $("#latitudeForMap option:selected").val(),
324  		long = $("#longitudeForMap option:selected").val(),
325  		size = $("#sizeForMap option:selected").val(),
326  		category = $("#categoryListForMap option:selected").val() ;
327 	for (index = 0; index < ARV.filteredData.data.length; index++) {
328 		dataItem = ARV.filteredData.data[index];
329 		latValue = parseInt(dataItem[lat],10);
330 		longValue = parseInt(dataItem[long],10);
331 		sizeValue = parseInt(dataItem[size] || 10 ,10);
332 		if (isNaN(latValue) || isNaN(longValue) || isNaN(sizeValue)) {
333 			$("#dialog-illegal-data").dialog("open");
334 			return undefined;
335 		}
336 		obj = {
337 			lat: latValue,
338 			lon: longValue,
339 			size: sizeValue,
340 			category: dataItem[category] || " "
341 		};
342 		dataArr.push(obj);
343 	}
344 	ARV.defaultData.MapGraph = dataArr;
345 };

347 /**
348  * The function calls the update data method for all the types for the graph viz single series, multiseries,
349  * bubble chart. 
350  */
351 ARV.updateDataForVisualizations = function(){
352 	if(ARV.filteredData.data.length === 0){
353 		$("#chart").html("Select some fields to view data");
354 		return;
355 	}
356     var noOfSelected = ARV.getNumberOfSelectedMeasureAxis();
357     try {
358         if (noOfSelected === 1) {
359             ARV.modifyDataForSingleSeries();
360         }
361         if (noOfSelected > 1) {
362             ARV.modifyDataForSingleSeries();
363             ARV.updateDataForMultiSeries();
364             if (noOfSelected === 3) {
365                 ARV.updateDataForBubbleChart();
366             }
367         }
368     }
369     catch (e) {
370         alert(e);
371     }
372     if("Map" === $("#chartTypes option:selected").attr("value")){
373     	ARV.updateDataForMap();
374     }
375     ARV.createDataForTable();
376     ARV.refreshTable();
377     ARV.refreshGraph();
378 };
