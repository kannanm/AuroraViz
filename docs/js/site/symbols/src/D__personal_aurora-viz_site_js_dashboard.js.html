 82 ARV.addTextComponents = function(componentClass,type,oldComponent,position){
 83 	var component;

 94 		'submit' : 'Done',

 96 		'remove' : 'Delete',
 97 		'toolTip': 'Double Click to edit',
 98 		'removeHandler': function(){
 99 							$(this).parent().remove();

101 		'type': type
102 	});
103 	var div = $("<div/>").addClass("draggable").addClass("element-container").append(component);
104 	if(position){
105 		ARV.setPosition(div, position);
106 		ARV.setPosition(component, ARV.defaultPosition);
107 		$(component).css("position","relative");
108 	}
