364 /**
365  * Adds the text component i.e. heading, label or a description field
366  * 
367  * @param componentClass
368  *            {string} The class of the component which can be label, heading or
369  *            description
370  * @param type
371  *            {string} type of the component which can be textbox or text area
372  * @param [oldComponent]
373  *            {Object} Optional field if given it means that the the new
374  *            component to be added will be the clone of the
375  *            <tt>oldComponent</tt>
376  * @param [position]
377  *            {Object} The position of the <tt>oldComponent</tt>. So the new
378  *            component to be added will be added in the same position
379  */
380 ARV.Dashboard.addTextComponents = function(componentClass, type, oldComponent, position) {
381 	var component, dashNS = ARV.Dashboard;
382 	if (oldComponent) {
383 		component = $(oldComponent).clone().addClass("component");
384 		oldComponent.remove();
385 	} else {
386 		component = dashNS.getComponent[componentClass]();
387 	}
388 	$(component).inLineEditor({
389 		'event' : 'dblclick',
390 		'onblur' : 'save',
391 		'submit' : 'Done',
392 		'cancel' : 'Undo',
393 		'remove' : 'Delete',
394 		'toolTip' : 'Double Click to edit',
395 		'removeHandler' : function() {
396 			$(this).parent().remove();
397 		},
398 		'type' : type
399 	});
400 	var div = $("<div/>").addClass("draggable").addClass("element-container").append(component);
401 	if (position) {
402 		dashNS.setPosition(div, position);
403 		dashNS.setPosition(component, dashNS.defaultPosition);
404 		$(component).css("position", "relative");
405 	}
406 	$("#dashboardEditor").append(div);
407 	$(".draggable").draggable({
408 		containment : "parent"
409 	});
410 
411 };

436 /**
437  * Inserts the dom elements for editing options. Checks if the selected element
438  * is a report or not and then populates the editing options accordingly
439  */
440 ARV.Dashboard.populateOptions = function() {
441 	var dashNS = ARV.Dashboard;
442 	var isReport = $(dashNS.selectedElement).is(".report");
443 	var html = isReport ? dashNS.options.report : dashNS.options.text;
444 	$("#options").html(html);
445 	$("button").button();
446 	ARV.Utility.initializeColorPicker(ARV.Utility.colorPickerCB.dashboard);
447 	ARV.Utility.createSliders(ARV.Utility.createSlidersCB.dashboard);
448 };

450 /**
451  * Sets the values in the editing options equal to the property values of the
452  * selected element
453  */
454 ARV.Dashboard.setOptionsValues = function() {
455 	var dashNS = ARV.Dashboard;
456 	var elem = dashNS.selectedElement;
457 	var color = ARV.Utility.rgb2hex($(elem).css("color"));
458 	var size = parseInt($(elem).css("font-size"), 10);
459 	var angle = ARV.Utility.matrixToDegree($(elem).css("-moz-transform"));
460 	$("#textColor").val(color);
461 	$("#textSize").val(size);
462 	$("#textSizeSlider").slider("value", size);
463 	$("#textRotateAngle").val(angle);
464 	$("#textRotateAngleSlider").slider("value", angle);
465 };

566 /**

568  */
569 ARV.Dashboard.initDashboard = function() {
570 	var dashNS = ARV.Dashboard;
571 	// initialize menu bar with jquery ui
572 	$("#addReports").menubar();
573 
574 	// initialize button with jquery ui
575 	$("button").button();
576 
577 	// Click handle if the add TextField button is clicked
578 	$("#addTextField").click(function() {
579 		dashNS.addTextComponents(dashNS.components.label, "textBox");
580 	});
581 
582 	// Click handle if the add Heading button is clicked
583 	$("#addHeading").click(function() {
584 		dashNS.addTextComponents(dashNS.components.heading, "textBox");
585 	});
586 
587 	// Click handle if the add Description button is clicked
588 	$("#addDesciption").click(function() {
589 		dashNS.addTextComponents(dashNS.components.description, "textArea");
590 	});
591 
592 	// Click handle if the add Report button is clicked
593 	$("#addReports li ul li").click(function() {
594 		var containerID = "report" + (dashNS.noOfReports);
595 		var id = $(this).children(":first").attr("id");
596 		var svgContainer = $("<div/>").addClass("component").attr("id", containerID).addClass("report");
597 		var div = $("<div/>").addClass("element-container").addClass("draggable").append(svgContainer);
598 		var graphType = dashNS.GraphData[id].graphType;
599 		$("#dashboardEditor").append(div);
600 		ARV.CG.showGraph(graphType, containerID, dashNS.GraphData[id]);
601 		$(".draggable").draggable({
602 			containment : "parent"
603 		});
604 		dashNS.originalGraphData[containerID] = {
605 			height : div.height(),
606 			width : div.width(),
607 			graphDef : dashNS.GraphData[id]
608 		};
609 		dashNS.setReportResizable();
610 		dashNS.noOfReports += 1;
611 	});
612 
613 	// Click handle if any of the components is clicked. It stylize it and adds
614 	// the options menu
615 	$(".component").live("click", function() {
616 		dashNS.removeSelectedElemStyling();
617 		$(this).addClass("selected");
618 		dashNS.setSelected($(this));
619 		dashNS.populateOptions();
620 		dashNS.setOptionsValues();
621 	});
622 
623 	// Makes the components editable if Edit button is clicked
624 	$("#editBtn").click(function() {
625 		var index = 0;
626 		for (index = 0; index < dashNS.editableTextElements.length; index++) {
627 			$("." + dashNS.editableTextElements[index]).each(function() {
628 				var componentClass = dashNS.editableTextElements[index];
629 				var type = componentClass === dashNS.components.description ? "textArea" : "textBox";
630 				dashNS.addTextComponents(componentClass, type, $(this), $(this).position());
631 			});
632 		}
633 		$(".report").each(function() {
634 			var div = $("<div/>").addClass("element-container").addClass("draggable").append($(this));
635 			$(this).addClass("component");
636 			$("#dashboardEditor").append(div);
637 			dashNS.setPosition(div, $(this).position());
638 			dashNS.setPosition($(this), dashNS.defaultPosition);
639 			$(this).css("position", "relative");
640 			$(".draggable").draggable({
641 				containment : "parent"
642 			});
643 			dashNS.setReportResizable();
644 		});
645 		// TODO stylize disabled button
646 		$(this).attr("disabled", "disabled");
647 	});
648 
649 	$(".delComponentBtn").live("click", function() {
650 		var parent = $(this).parent();
651 		$(parent).remove();
652 
653 	});
