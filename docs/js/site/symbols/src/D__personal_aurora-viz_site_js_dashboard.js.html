304 ARV.addTextComponents = function(componentClass, type, oldComponent, position) {
305     var component;
306     if (oldComponent) {
307         component = $(oldComponent).clone().addClass("component");
308         oldComponent.remove();
309     }
310     else {
311         component = ARV.getComponent[componentClass]();
312     }
313     $(component).inLineEditor({
314         'event': 'dblclick',
315         'onblur': 'save',
316         'submit': 'Done',
317         'cancel': 'Undo',
318         'remove': 'Delete',
319         'toolTip': 'Double Click to edit',
320         'removeHandler': function() {
321             $(this).parent().remove();
322         },
323         'type': type
324     });
325     var div = $("<div/>").addClass("draggable").addClass("element-container").append(component);
326     if (position) {
327         ARV.setPosition(div, position);
328         ARV.setPosition(component, ARV.defaultPosition);
329         $(component).css("position", "relative");
330     }
331     $("#dashboardEditor").append(div);
332     $(".draggable").draggable({
333         containment: "parent"
334     });
335 
336 };

359 ARV.populateOptions = function() {
360     var isReport = $(ARV.selectedElement).is(".report");
361     var html = isReport ? ARV.options.report : ARV.options.text;
362     $("#options").html(html);

367 ARV.setOptionsValues = function() {
368     var color = ARV.rgb2hex($(ARV.selectedElement).css("color"));
369     var size = parseInt($(ARV.selectedElement).css("font-size"), 10);
370     var angle = ARV.matrixToDegree($(ARV.selectedElement).css("-moz-transform"));
371     $("#textColor").val(color);
372     $("#textSize").val(size);
373     $("#textSizeSlider").slider("value", size);
374     $("#textRotateAngle").val(angle);
375     $("#textRotateAngleSlider").slider("value", angle);
376 };

424 ARV.initDashboard = function() {
425     //initialize menu bar with jquery ui
426     $("#addReports").menubar();
427 
428     //initialize button with jquery ui
429     $("button").button();
430 
431     //Click handle if the add TextField button is clicked
432     $("#addTextField").click(function() {
433         ARV.addTextComponents(ARV.components.label, "textBox");
434     });
435 
436     //Click handle if the add Heading button is clicked
437     $("#addHeading").click(function() {
438         ARV.addTextComponents(ARV.components.heading, "textBox");
439     });
440 
441     //Click handle if the add Description button is clicked
442     $("#addDesciption").click(function() {
443         ARV.addTextComponents(ARV.components.description, "textArea");
444     });
445 
446     //Click handle if the add Report button is clicked
447     $("#addReports li ul li").click(function() {
448         var containerID = "report" + (ARV.noOfReports);
449         var id = $(this).children(":first").attr("id");
450         var svgContainer = $("<div/>").addClass("component").attr("id", containerID).addClass("report");
451         var div = $("<div/>").addClass("element-container").addClass("draggable").append(svgContainer);
452         var graphType = ARV.GraphData[id].graphType;
453         $("#dashboardEditor").append(div);
454         ARV.showGraph(graphType, containerID, ARV.GraphData[id]);
455         $(".draggable").draggable({
456             containment: "parent"
457         });
458         ARV.originalGraphData[containerID] = {
459             height: div.height(),
460             width: div.width(),
461             graphDef: ARV.GraphData[id]
462         };
463         ARV.setReportResizable();
464         ARV.noOfReports += 1;
465     });
466 
467     //Click handle if any of the components is clicked. It stylize it and adds the options menu
468     $(".component").live("click", function() {
469         ARV.removeSelectedElemStyling();
470         $(this).addClass("selected");
471         ARV.setSelected($(this));
472         ARV.populateOptions();
473         ARV.setOptionsValues();
474     });
475 
476     //Makes the components editable if Edit button is clicked
477     $("#editBtn").click(function() {
478         var index = 0;
479         for (index = 0; index < ARV.editableTextElements.length; index++) {
480             $("." + ARV.editableTextElements[index]).each(function() {
481                 var componentClass = ARV.editableTextElements[index];
482                 var type = componentClass === ARV.components.description ? "textArea" : "textBox";
483                 ARV.addTextComponents(componentClass, type, $(this), $(this).position());
484             });
485         }
486         $(".report").each(function() {
487             var div = $("<div/>").addClass("element-container").addClass("draggable").append($(this));
488             $(this).addClass("component");
489             $("#dashboardEditor").append(div);
490             ARV.setPosition(div, $(this).position());
491             ARV.setPosition($(this), ARV.defaultPosition);
492             $(this).css("position", "relative");
493             $(".draggable").draggable({
494                 containment: "parent"
495             });
496             ARV.setReportResizable();
497         });
498         //TODO stylize disabled button
499         $(this).attr("disabled", "disabled");
500     });
501 
502 
503     $(".delComponentBtn").live("click", function() {
504         var parent = $(this).parent();
505         $(parent).remove();
506 
507     });
508 
509     //Makes all the components non editable if the save button is clicked
510     $("#saveBtn").click(function() {
511         $(".component").each(function() {
512             var position = $(this).parent().position();
513             var a = $(this).clone();
514             a.removeClass("component");
515             a.css("position", "absolute");
516             a.css("top", position.top);
517             a.css("left", position.left);
518             a.attr("title", "Click on the Edit Button to Edit");
519             $(this).parent().parent().append(a);
520             $(this).parent().remove();
521         });
522         $("#editBtn").removeAttr("disabled");
523         $(".report").each(function() {
524             var containerID = $(this).attr("id");
525             var graphDef = ARV.originalGraphData[containerID].graphDef;
526             ARV.showGraph(graphDef.graphType, containerID, graphDef);
527         });
528         ARV.removeSelectedElemStyling();
529     });

532     });
