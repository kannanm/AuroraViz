  6 /**
  7  * Populates the category filter with all the category values of the selected category axis
  8  * @param {String} id the id of the category dom element from where the selected category axis is to be taken 
  9  */
 10 ARV.addAllCategories = function(id){
 11 	var selected = $("#"+id+ " option:selected");
 12 	var selectedVal = $(selected).val();
 13 	var categories = $("#categoryValues");
 14 	$(categories).empty();
 15 	var i = 0;
 16 	var option;
 17 	var originalData = ARV.dataJSON.data;
 18 	for(i=0;i<originalData.length;i++){
 19 		 var dataItem = originalData[i]
 20 		 option = $('<option />').val(dataItem[selectedVal]).append(dataItem[selectedVal]).attr("selected", "selected");
 21 		 $(categories).append(option);
 22 	}
 23 	$(categories).trigger("liszt:updated");
 24 };

 26 /**
 27  * Set the value attribute "selected" equal to "selected" for the input element <tt>elem</tt> 
 28  * @param elem The element for which all the options have to be set to selected
 29  */
 30 ARV.setAllSelected = function(elem){
 31 	$(elem).children("option").each(function(){

 47 /**
 48  * Calculate and return the data filtered after applying filters based selected measure filter values 
 49  * @param originalData {Object} The data to be filtered
 50  * @returns {Object} The filtered data 
 51  */
 52 ARV.getMeasureFilteredData = function(originalData){
 53 	var filteredData = originalData;
 54 	$(".measure-slider").each(function(){
 55 		var min = $(this).slider("values",0);
 56 		var max = $(this).slider("values",1);
 57 		var measureFeild = $(this).attr("data-measureField");
 58 		filteredData = jlinq.from(filteredData).betweenEquals(measureFeild,min,max).select();
 59 	});
 60 	return filteredData;
 61 };

 63 /**
 64  * Calculate and return the data filtered after applying filters based selected category filter values 
 65  * @param originalData {Object} The data to be filtered
 66  * @returns {Object} The filtered data 
 67  */
 68 ARV.getCategoryFilteredData = function(originalData){
 69 	var selected = $("#categoryValues option:selected");
 70 	var category = $("#categoryAxisList option:selected").val();
 71 	var filteredData = [];
 72 	var i = 0;
 73 	for(i=0;i<selected.length;i++){
 74 		var categoryValue = $(selected[i]).val();
 75 		var dataItems = jlinq.from(originalData).equals(category,categoryValue).select();
 76 		var j =0;
 77 		for(j=0;j<dataItems.length;j++){
 78 			filteredData.push(dataItems[j]);
 79 		}
 80 		
 81 	}
 82 	return filteredData;
 83 };

 85 /**
 86  * Filter the data and update the data to show the visualization
 87  */
 88 ARV.filterData = function(){
 89 	var categoryFilteredData = ARV.getCategoryFilteredData(ARV.dataJSON.data);
 90 	ARV.filteredData.data = ARV.getMeasureFilteredData(categoryFilteredData);
 91 	ARV.updateDataForVisualizations();
 92 };

107 /**
108  * Create range slider for the measure fields
109  * @param id  {Number} random id is created to append to the slider dom element id. Thus the id of 
110  * the slider would be <tt>id</tt>+"Slider" 
111  * @param min {Number} The min value of the measure field
112  * @param max {Number} The max value of the measure field
113  * @param measureFeild {String} The measure field
114  */
115 ARV.createRangeSlider = function(id,min,max,measureFeild){
116 	$( "#"+id +"Slider").slider({
117 		range: true,
118 		min: min,
119 		max: max,
120 		values: [ min, max ],
121 		stop: function( event, ui ) {
122 			var minVal = ui.values[ 0 ];
123 			var maxVal = ui.values[ 1 ];
124 			$( "#"+id ).val(minVal + " - " + maxVal );
125 			ARV.filterData();
126 		}
127 	});
128 	$(  "#"+id ).val( $( "#"+id +"Slider" ).slider( "values", 0 ) +
129 			" - " + $( "#"+id +"Slider" ).slider( "values", 1 ) );
130 };

132 /**
133  * Finds the minimum value of the <tt>measureField</tt> from <tt>ARV.filteredData.data</tt>
134  * @param measureField {String} The measure field
135  * @returns {Number} The minimum value of the <tt>measureField</tt>
136  */
137 ARV.getMinOfMeasureField = function(measureField){
138 	var minValObj = jlinq.from(ARV.filteredData.data).min(measureField);
139 	return minValObj[measureField];
140 };
141 /**
142  * Finds the maximum value of the <tt>measureField</tt> from <tt>ARV.filteredData.data</tt>
143  * @param measureField {String} The measure field
144  * @returns {Number} The maximum value of the <tt>measureField</tt>
145  */
146 ARV.getMaxOfMeasureField = function(measureField){
147 	var minValObj = jlinq.from(ARV.filteredData.data).max(measureField);
148 	return minValObj[measureField];
149 };

151 /**
152  * Creates the measure filteres according to the measure axis selected
153  */
154 ARV.addMeasureFilters = function(){
155 	$("#measureFilter").html("");
156 	var selected = ARV.selectedMeasureAxis;
157 	var i = 0;
158 	for(i=0;i<selected.length;i++){
159 		var measureFeild = selected[i];
160 		var id = "measure"+i;
161 		var label = $("<label/>").html(measureFeild);
162 		var slider = $("<div/>").addClass("range-slider").attr("id",id+"Slider").addClass("measure-slider").attr("data-measureField",measureFeild);
163 		var text = $("<input/>").attr("size","3").attr("type","text").attr("id",id).attr("disabled","disabled").addClass("range-val");
164 		$("#measureFilter").append(label);
165 		$("#measureFilter").append(slider).append(text);
166 		$("#measureFilter").append("<br/>");
167 		min = ARV.getMinOfMeasureField(measureFeild);
168 		max = ARV.getMaxOfMeasureField(measureFeild);
169 		ARV.createRangeSlider(id,min,max,measureFeild);
170 	}
171 	
172 };

201 /**
