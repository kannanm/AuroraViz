 10 ARV.addAllCategories = function(id){
 11 	var selected = $("#"+id+ " option:selected");
 12 	var selectedVal = $(selected).val();
 13 	var categories = $("#categoryValues");
 14 	$(categories).empty();
 15 	var i = 0;
 16 	var option;
 17 	var originalData = ARV.dataJSON.data;
 18 	for(i=0;i<originalData.length;i++){
 19 		 var dataItem = originalData[i]
 20 		 option = $('<option />').val(dataItem[selectedVal]).append(dataItem[selectedVal]).attr("selected", "selected");
 21 		 $(categories).append(option);
 22 	}
 23 	$(categories).trigger("liszt:updated");
 24 };
 25 ARV.setAllSelected = function(elem){

 37 ARV.getMeasureFilteredData = function(originalData){
 38 	var filteredData = originalData;
 39 	$(".measure-slider").each(function(){
 40 		var min = $(this).slider("values",0);
 41 		var max = $(this).slider("values",1);
 42 		var measureFeild = $(this).attr("data-measureField");
 43 		filteredData = jlinq.from(filteredData).betweenEquals(measureFeild,min,max).select();
 44 	});
 45 	return filteredData;
 46 };
 47 ARV.getCategoryFilteredData = function(originalData){
 48 	var selected = $("#categoryValues option:selected");
 49 	var category = $("#categoryAxisList option:selected").val();
 50 	var filteredData = [];
 51 	var i = 0;
 52 	for(i=0;i<selected.length;i++){
 53 		var categoryValue = $(selected[i]).val();
 54 		var dataItems = jlinq.from(originalData).equals(category,categoryValue).select();
 55 		var j =0;
 56 		for(j=0;j<dataItems.length;j++){
 57 			filteredData.push(dataItems[j]);
 58 		}
 59 		
 60 	}
 61 	return filteredData;
 62 };
 63 ARV.filterData = function(){
 64 	var categoryFilteredData = ARV.getCategoryFilteredData(ARV.dataJSON.data);
 65 	ARV.filteredData.data = ARV.getMeasureFilteredData(categoryFilteredData);
 66 	ARV.updateDataForVisualizations();
 67 };
 68 $("#categoryValues").chosen().change(ARV.filterData);
 69 $("#allCategories").change(function(){
 70 	if($(this).attr("checked")){
 71 		ARV.setAllSelected($("#categoryValues"));
 72 	}else{
 73 		ARV.setAllDeselected($("#categoryValues"));
 74 	}
 75 	$("#categoryValues").trigger("liszt:updated");
 76 	ARV.filterData();
 77 });
 78 ARV.createRangeSlider = function(id,min,max,measureFeild){
 79 	$( "#"+id +"Slider").slider({
 80 		range: true,
 81 		min: min,
 82 		max: max,
 83 		values: [ min, max ],
 84 		stop: function( event, ui ) {
 85 			var minVal = ui.values[ 0 ];
 86 			var maxVal = ui.values[ 1 ];
 87 			$( "#"+id ).val(minVal + " - " + maxVal );
 88 			ARV.filterData();
 89 		}
 90 	});
 91 	$(  "#"+id ).val( $( "#"+id +"Slider" ).slider( "values", 0 ) +
 92 			" - " + $( "#"+id +"Slider" ).slider( "values", 1 ) );
 93 };
 94 ARV.getMinOfMeasureField = function(measureField){
 95 	var minValObj = jlinq.from(ARV.filteredData.data).min(measureField);
 96 	return minValObj[measureField];
 97 };
 98 ARV.getMaxOfMeasureField = function(measureField){
 99 	var minValObj = jlinq.from(ARV.filteredData.data).max(measureField);
100 	return minValObj[measureField];
101 };

103 ARV.addMeasureFilters = function(){
104 	$("#measureFilter").html("");
105 	var selected = ARV.selectedMeasureAxis;
106 	var i = 0;
107 	for(i=0;i<selected.length;i++){
108 		var measureFeild = selected[i];
109 		var id = "measure"+i;
110 		var label = $("<label/>").html(measureFeild);
111 		var slider = $("<div/>").addClass("range-slider").attr("id",id+"Slider").addClass("measure-slider").attr("data-measureField",measureFeild);
112 		var text = $("<input/>").attr("size","3").attr("type","text").attr("id",id).attr("disabled","disabled").addClass("range-val");
113 		$("#measureFilter").append(label);
114 		$("#measureFilter").append(slider).append(text);
115 		$("#measureFilter").append("<br/>");
116 		min = ARV.getMinOfMeasureField(measureFeild);
117 		max = ARV.getMaxOfMeasureField(measureFeild);
118 		ARV.createRangeSlider(id,min,max,measureFeild);
119 	}
120 	
121 };
