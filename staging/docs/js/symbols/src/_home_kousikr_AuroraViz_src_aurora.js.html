  3 Array.prototype.max = Array.prototype.max || function (){
  4 	var max = this[0];
  5 	var len = this.length;
  6 	var i;
  7 	for (i = 1; i < len; i++){
  8 		if (this[i] > max){
  9 			max = this[i];
 10 		}
 11 	}
 12 	return max;
 13 };

 36 	function g () {
 37 	}
 38 	g.prototype = f.prototype || f;
 39 	return new g();

 42 /**
 43  * @author Aditya Gaur
 44  * Adds the javascript file specified by the <tt>jsPath</tt>,
 45  *         at the position <tt>pos</tt>.
 46  * @param {string}
 47  *            [jsPath] the path of the JS file to be included.
 48  * @param {string}
 49  *            [pos](optional)the dom element in which the script tag to include
 50  *            js file will appear. If unspecified the js file will be included
 51  *            in head.
 52  */
 53 AR.addJavascript = function (jsPath, pos, dataPath, callback) {
 54 	var position = pos || "head";
 55 	var posElem = document.getElementsByTagName(position)[0];
 56 	var scriptTag = document.createElement('script');
 57 	scriptTag.setAttribute('type', 'text/javascript');
 58 	scriptTag.setAttribute('src', jsPath);
 59 	if (dataPath) {
 60 		scriptTag.setAttribute('data-path', dataPath);
 61 	}
 62 	posElem.appendChild(scriptTag);
 63 	return scriptTag;
 64 };

 89 /**
 90  * The basefunction for the graph initilization. 
 91  * Initializes the panel and sets parentDimensions and property values
 92  * @params {json} 
 93  * 			graphDef is the current graph Dimension
 94  * This method should be the first method that should be called when creating any kind of graph.
 95  * Creates the panel, sets its dimensions, labels and property values and returns. 
 96  */
 97 AR.Graph = function (graphDef) {
 98 	var self = this;
 99 	self.__graphDef = graphDef;
100 	self._pos = undefined;
101 	self._dimension = {
102 	height : 400,
103 	width : 400,
104 	left : 40,
105 	right : 20,
106 	bottom : 60,
107 	top : 20
108 	};
109 	self.__captionLabel = undefined;
110 	self.__xAxisName = undefined;
111 	self.__yAxisName = undefined;
112 	self._horRules = undefined;
113 	self._verticalRules = undefined;
114 	self._panel = new pv.Panel();
115 	self._panel.left(self._dimension.left).right(self._dimension.right).bottom(self._dimension.bottom).top(self._dimension.top).strokeStyle("black");
116 	self._xAxis = self._panel.add(pv.Rule).bottom(0);
117 	self._yAxis = self._panel.add(pv.Rule).left(0);
118 	this.properties.forEach(function (property) {
119 		var upcasedProp = property.substring(0, 1).toUpperCase() + property.substring(1);
120 		if (graphDef[property]) {
121 			self["set" + upcasedProp](graphDef[property]);
122 		}
123 	});
124 };
125 AR.Graph.prototype.properties = ["width", "height", "caption", "xAxisName", "yAxisName", "borderColor", "fillColor"];

209 /**
210  * Sets horizontal rules in a graph.
211  * Rules are nothing but lines that divide the graph to form a grid like structure
212  * @params{integer, scaleType}
213  * 			It takes the scaleType (Linear/Ordinal etc) and an integer value that denotes the number of rules to be set
214  * This method setHorRules permenantly sticks with the AR.Graph object as it is prototyped.
215  * This can be used by any AR.Graph type object
216  */ 
217 AR.Graph.prototype.setHorRules = function (maxValue, scaleType) {
218 	var self = this;
219 	var y;
220 	if (!self._horRules) {
221 		y = pv.Scale[scaleType](0, maxValue).range(0, self._dimension.height - 40);
222 		self._horRules = self._panel.add(pv.Rule);
223 		self._horRules.data(y.ticks()).bottom(y).strokeStyle(function (d) {
224 			return (d ? "#eee" : "#000");
225 		}).anchor("left").add(pv.Label).text(y.tickFormat);
226 	} else {
227 		y = pv.Scale.linear(0, maxValue).range(0, self._dimension.height - 40);
228 		self._horRules.bottom(y);
229 	}
230 };
