  2 /**
  3  * The basefunction for the Stacked Area Graph initialization. Need x and y co-ordinates.  
  4  * Initializes the StackedAreaGraph. Sets the data
  5  * @params {Object}
  6  * 			[graphDef] graphDef object supplied by the User for which the graph needs to be plotted. It contains graph properties and the data
  7  * @params {Object}
  8  * 			[parentDimension] parentDimension object that will indicate the dimensions of AR.Graph
  9  * @params {Object}
 10  * 			[panel] A panel object indicating the Graph Panel in which the current Area graph will be displayed. 
 11  * This is a base class and is called by different Area implementations. 
 12  */
 13 AR.stackedarea = function(parentDimension, panel, graphDef){
 14 	var self = this;
 15 	// dataValues will contain the whole x,y,z values in form of an array. 
 16 	var dataValues = AR.Utility.getTwoDimensionData(graphDef.data);
 17 	//self._stackedArea = panel.add(pv.Layout.Stack);
 18 	self._stackedArea = panel.add(pv.Area);
 19 	var datum = graphDef.data;
 20 	//self._stackedArea.layers(datum);
 21 	self._stackedArea.data(datum);
 22 
 23 	var setX = function(parentDimension){
 24 		var maxVal = AR.Utility.getSingleDimensionData(graphDef.data,AR.Utility.Dimension.x).max();
 25 		var xScale = pv.Scale.linear(0, maxVal).range(0, parentDimension.width-40); 
 26 		//var xScale = pv.Scale.linear(graphDef.data, function(d) d.x).range(0, parentDimension.width); 
 27 		self._stackedArea.left(function(d){
 28 			return xScale(d.x);
 29 		});
 30 	};
 31 	
 32 	var setY = function(parentDimension){
 33 		var maxVal = AR.Utility.getSingleDimensionData(graphDef.data,AR.Utility.Dimension.y).max();
 34 		var yScale = pv.Scale.linear(0, maxVal).range(0, parentDimension.height-40);
 35 		self._stackedArea.height(function(d){
 36 			return yScale(d.y)
 37 		});
 38 	};
 39 	
 40 	var setXandY = function(parentDimension){
 41 			setX(parentDimension);
 42 			setY(parentDimension);
 43 	};
 44 	
 45 	var setBottom = function(){
 46 		self._stackedArea.bottom(0);
 47 	};
 48 	
 49 	var setFillStyle = function(){
 50 		self._stackedArea.fillStyle("rgb(121,173,210)");
 51 	};
 52 	
 53 	var setAnchor = function(){
 54 		self._stackedArea.anchor("top").add(pv.Line).lineWidth(3);
 55 	};
 56 	
 57 		
 58 	self.setStackAttributes = function(){
 59 			var self = this;
 60 			setBottom();
 61 			setXandY(parentDimension);	
 62 			setFillStyle();
 63 			setAnchor();
 64 			if(graphDef.segmented == true){
 65 				self._stackedArea.segmented(true);
 66 			}
 67 			if(graphDef.interpolate == true){
 68 				self._stackedArea.interpolate("step-after");
 69 			}
 70 	};
 71 	
 72 	
 73 	self.setStackAttributes();
 74 	//self._stackedArea.layer.add(pv.Area);		
 75 };

 77 /**
 78  * The basefunction for the Stacked Area Graph initialization. MultiGraph this is! 
 79  * Initializes the StackedAreaGraph. Sets the data
 80  * @params {Object}
 81  * 			[graphDef] graphDef object supplied by the User for which the graph needs to be plotted. It contains graph properties and the data
 82  * @params {Object}
 83  * 			[parentDimension] parentDimension object that will indicate the dimensions of AR.Graph
 84  * @params {Object}
 85  * 			[panel] A panel object indicating the Graph Panel in which the current Area graph will be displayed. 
 86  * This is a base class and is called by different Area implementations. 
 87  */
 88 AR.multistackedarea = function(parentDimension, panel, graphDef, data){
 89 	// TODO  Multiple area graphs in the same canvas
 90 	var self = this;
 91 	self._panel = panel.add(pv.Layout.Stack);
 92 	self._panel.layers(data);
 93 	
 94 	var dataArray = new Array();
 95 		for(i=0; i<graphDef.dataset.length; i++){
 96 			var max = AR.Utility.getSingleDimensionData(graphDef.dataset[i].data, AR.Utility.Dimension.y).max();
 97 			var maxMap = {"value" : max};
 98 			dataArray.push(maxMap);
 99 		}
100 	var yMax = AR.Utility.findMax(dataArray);
101 	
102 	var XdataArray = new Array();
103 	for(i=0; i<graphDef.dataset.length; i++){
104 		var max = AR.Utility.getSingleDimensionData(graphDef.dataset[i].data, AR.Utility.Dimension.x).max();
105 		var maxMap = {	"label" : "data",
106 						"value" : max};
107 		XdataArray.push(maxMap);
108 	}
109 	var xMax = AR.Utility.findMax(XdataArray);
110 	
111 	var setXvalue = function(parentDimension){
112 		var xScale = pv.Scale.linear(0, xMax).range(0, parentDimension.width-40);
113 		self._panel.x(function(d){
114 			return xScale(d.x);
115 		});
116 	};
117 	
118 	var setYvalue = function(parentDimension){
119 		var yScale = pv.Scale.linear(0, yMax).range(0, parentDimension.height-40);
120 		self._panel.y(function(d){
121 			return yScale(d.y);
122 		});
123 	};
124 	
125 	var setBottom = function(){
126 		self._panel.bottom(0);
127 	};
128 	
129 	var setFillStyle = function(){
130 		self._panel.fillStyle("rgb(121,173,210)");
131 	};
132 	
133 	self.setDataValues = function(){
134 		setXvalue(parentDimension);
135 		setYvalue(parentDimension);
136 		setBottom();
137 		self._panel.layer.add(pv.Area);
138 	}
139 	
140 	self.setDataValues();
141 };

148  * @extends AR.Graph
149  */
150 AR.StackedAreaGraph = function(graphDef){
151 
152 /*	var data = pv.range(4).map(function() {
153     return pv.range(0, 10, .1).map(function(x) {
154         return {x: x, y: Math.sin(x) + Math.random() * .5 + 2};
155       });
156   });
157 */

159    
160 	var self = this;
161 	var stackedArea;
162 	AR.Graph.apply(self,[graphDef]);
163 	
164 	var setRules = {
165 	"v" : function () {
166 		if(graphDef.data){
167 			var maxVal = AR.Utility.getSingleDimensionData(graphDef.data,AR.Utility.Dimension.y).max();
168 			self.setHorRules(maxVal,AR.Utility.scale.linear);
169 		}else{
170 			var dataArray = new Array();

172 				var max = AR.Utility.getSingleDimensionData(graphDef.dataset[i].data, AR.Utility.Dimension.y).max();

174 				dataArray.push(maxMap);
175 			}
176 			self.setHorRules(AR.Utility.findMax(dataArray),AR.Utility.scale.linear);
177 		}	
178 	},
179 
180 	"h" : function () {
181 		if(graphDef.data){
182 			var maxVal = AR.Utility.getSingleDimensionData(graphDef.data,AR.Utility.Dimension.x).max();
183 			self.setVerticalRules(maxVal,AR.Utility.scale.linear);
184 		}else{
185 			var dataArray = new Array();
186 			for(i=0; i<graphDef.dataset.length; i++){
187 				var max = AR.Utility.getSingleDimensionData(graphDef.dataset[i].data, AR.Utility.Dimension.x).max();
188 				var maxMap = {	"label" : "data",
189 								"value" : max};
190 				dataArray.push(maxMap);
191 			}
192 			self.setVerticalRules(AR.Utility.findMax(dataArray), AR.Utility.scale.linear);
193 		}
194 	},
195 	
196 	"yes" : function () {
197 		if(graphDef.data){
198 			var xmaxVal = AR.Utility.getSingleDimensionData(graphDef.data,AR.Utility.Dimension.y).max();
199 			self.setHorRules(xmaxVal,AR.Utility.scale.linear);
200 			
201 			var ymaxVal = AR.Utility.getSingleDimensionData(graphDef.data,AR.Utility.Dimension.x).max();
202 			self.setVerticalRules(ymaxVal,AR.Utility.scale.linear);
203 		}else{
204 			var XdataArray = new Array();
205 			for(i=0; i<graphDef.dataset.length; i++){
206 				var max = AR.Utility.getSingleDimensionData(graphDef.dataset[i].data, AR.Utility.Dimension.x).max();
207 				var maxMap = {	"label" : "data",
208 								"value" : max};
209 				XdataArray.push(maxMap);
210 			}
211 			self.setVerticalRules(AR.Utility.findMax(XdataArray), AR.Utility.scale.linear);
212 			
213 			var YdataArray = new Array();
214 			for(i=0; i<graphDef.dataset.length; i++){
215 				var max = AR.Utility.getSingleDimensionData(graphDef.dataset[i].data, AR.Utility.Dimension.y).max();

218 			}
219 			self.setHorRules(AR.Utility.findMax(YdataArray),AR.Utility.scale.linear);
220 		}	
221 	}
222 	 
223 	};
224 	
225 	setRules[graphDef.grid || "yes"]();
226 	
227 	if(graphDef.dataset){
228 			var dataset = graphDef.dataset;
229 			var dataArray = new Array();
230 			for(i=0; i<graphDef.dataset.length; i++){
231 				dataArray.push(dataset[i].data);
232 			}
233 			stackedArea = new AR.multistackedarea(self._dimension, self._panel, graphDef, dataArray);
234 			//document.write(dataArray);
235 	}else
236 		stackedArea = new AR.stackedarea(self._dimension, self._panel, graphDef);

252 	};
