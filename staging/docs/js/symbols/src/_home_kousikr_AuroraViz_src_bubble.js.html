  4 /**
  5  * The basefunction for the Bubble initialization. 
  6  * Initializes the BubbleGraph. Sets the data
  7  * @params {Object}
  8  * 			[graphDef] graphDef object supplied by the User for which the graph needs to be plotted. It contains graph properties and the data
  9  * @params {Object}
 10  * 			[parentDimension] parentDimension object that will indicate the dimensions of AR.Graph
 11  * @params {Object}
 12  * 			[panel] A panel object indicating the Graph Panel in which the current Bubble graph will be displayed
 13  * This is a base class and is called by different Bubble implementations
 14  */
 15 AR.Bubble = function(parentDimension, panel, graphDef) {
 16 	var self = this;
 17 	var bubble = panel.add(pv.Dot);
 18 	bubble.data(AR.Utility.getMultiDimensionData(graphDef.data));
 19 	bubble.fillStyle("red");
 20 	var adjustBottom = function (parentDimension) {
 21 		var maxVal = AR.Utility.getSingleDimensionData(graphDef.data,AR.Utility.Dimension.y).max();
 22 		var bubbleBottom = pv.Scale.linear(0, maxVal).range(0, parentDimension.height - 40);
 23 		bubble.bottom(function (d) {
 24 			return bubbleBottom(d[1]);
 25 		});
 26 	};
 27 	var adjustLeft = function (parentDimension) {
 28 		var maxVal = AR.Utility.getSingleDimensionData(graphDef.data,AR.Utility.Dimension.x).max();
 29 		var bubbleLeft = pv.Scale.linear(0, maxVal).range(0, parentDimension.width - 30);
 30 		bubble.left(function (d) {
 31 			return (bubbleLeft(d[0]));
 32 		});
 33 	};
 34 	var adjustSize = function (parentDimension){
 35 		var maxVal = AR.Utility.getSingleDimensionData(graphDef.data,AR.Utility.Dimension.z).max();
 36 		var smallerDim = Math.min(parentDimension.width,parentDimension.height);
 37 		var bubbleRadius = pv.Scale.linear(0, maxVal).range(0, smallerDim/20);
 38 		bubble.shapeRadius(function (d){
 39 			return (bubbleRadius(d[2]));
 40 		});
 41 	};
 42 	bubble.title(function () {
 43 		return AR.Utility.getToolTipText(graphDef.data, this.index);
 44 	});
 45 	if (graphDef.toolTip && graphDef.toolTip === 1) {
 46 		bubble.event("mouseover", pv.Behavior.tipsy({
 47 		gravity : function () {
 48 			return ("s");
 49 		},
 50 		fade : true
 51 		}));
 52 	}
 53 	self.adjustPosition = function (parentDimension){
 54 		adjustBottom(parentDimension);
 55 		adjustLeft(parentDimension);
 56 		adjustSize(parentDimension);
 57 	};
 58 	self.adjustPosition(parentDimension);
 59 };

 61 /**
 62  * API to construct a Bubble Graph
 63  * @param {object}
 64  *            [graphDef] An object containing the graph properties and the data
 65  * @extends AR.Graph
 66  */
 67 AR.BubbleGraph = function(graphDef){
 68 	var self = this;
 69 	AR.Graph.apply(self, [graphDef]);
 70 	self.setHorRules(AR.Utility.getSingleDimensionData(graphDef.data, AR.Utility.Dimension.y).max(),AR.Utility.scale.linear);
 71 	self.setVerticalRules(AR.Utility.getSingleDimensionData(graphDef.data, AR.Utility.Dimension.x).max(),AR.Utility.scale.linear);
 72 	var bubbles = new AR.Bubble(self._dimension, self._panel, graphDef);
 73 	//TODO: add other function such as changing the pallete etc  
 74 	self.setWidth = function (width) {
 75 		AR.Graph.prototype.setWidth.call(self, width);
 76 		bubbles.adjustPosition(self._dimension);
 77 	};
 78 	self.setHeight = function (height) {
 79 		AR.Graph.prototype.setHeight.call(self, height);
 80 		bubbles.adjustPosition(self._dimension);
 81 	};
 82 };
 83 AR.BubbleGraph.prototype = AR.extend(AR.Graph);
